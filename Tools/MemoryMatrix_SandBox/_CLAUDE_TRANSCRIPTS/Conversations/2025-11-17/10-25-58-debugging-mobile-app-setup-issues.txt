```js
export class AppStateManager {
  constructor() {
    this.state = {
      screen: 'input',
      userInput: '',
      generatedPrompt: null,
      selectedSection: null,
      isGenerating: false
    };
    this.listeners = [];
  }

  subscribe(listener) {
    this.listeners.push(listener);
  }

  notify() {
    this.listeners.forEach(listener => listener(this.state));
  }

  setState(updates) {
    this.state = { ...this.state, ...updates };
    this.notify();
  }

  getCurrentScreen() {
    return this.state.screen;
  }

  setScreen(screen) {
    this.setState({ screen });
  }
}
```

### src/services/AnthropicAPI.js
```js
export class AnthropicAPI {
  constructor() {
    this.apiKey = import.meta.env.VITE_ANTHROPIC_API_KEY;
    this.endpoint = 'https://api.anthropic.com/v1/messages';
  }

  async sendMessage(systemPrompt, userMessage) {
    const response = await fetch(this.endpoint, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'x-api-key': this.apiKey,
        'anthropic-version': '2023-06-01'
      },
      body: JSON.stringify({
        model: 'claude-sonnet-4-20250514',
        max_tokens: 4096,
        system: systemPrompt,
        messages: [{ role: 'user', content: userMessage }]
      })
    });

    return await response.json();
  }
}
```

## Key Design Decisions

### 1. **No React Dependency**
- Pure vanilla JS with ES6 classes
- Event-driven updates
- Manual DOM manipulation for performance

### 2. **Component Pattern**
```js
class Component {
  render() { return HTMLElement }
  attach(container) { /* add listeners */ }
  detach() { /* cleanup */ }
}
```

### 3. **State Flow**
```
User Action → Controller → StateManager.setState() 
→ StateManager.notify() → Screen.render()
```

### 4. **Section Detection**
Uses keyword matching (same as V7):
- Always include: Role & Context, Task & Objective
- Conditional: Process, Rules, Output, Examples
- Minimum 3 sections

### 5. **Double-Click Pattern**
```js
// First click
clickedOnce = sectionId
setTimeout(() => clickedOnce = null, 2000)

// Second click (if within 2s)
if (clickedOnce === sectionId) drillDown()
```
