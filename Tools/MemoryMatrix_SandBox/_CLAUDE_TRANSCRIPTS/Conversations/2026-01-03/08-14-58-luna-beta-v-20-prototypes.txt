## Analysis Engine Sketch

```python
# analysis/detector.py

ASSISTANT_PHRASES = [
    "sure thing", "of course", "certainly", "i'd be happy to",
    "feel free to", "let me know", "how can i help",
    "is there anything", "would you like me to",
]

CONFABULATION_MARKERS = [
    "i remember", "i recall", "from what i remember",
    "it was a while ago but", "i think i can piece together",
]

THIRD_PERSON_MARKERS = [
    "as luna", "luna is", "luna was", "we were designed",
    "the ai", "this ai", "i'm designed to",
]

@dataclass
class ResponseAnalysis:
    assistant_leak_score: float      # 0-1
    verbosity_score: float           # 0-1  
    voice_match_score: float         # 0-1
    flags: List[str]                 # Detected issues
    
def analyze_response(response: str, expected_length: int = 30) -> ResponseAnalysis:
    text_lower = response.lower()
    flags = []
    
    # Assistant leak detection
    assistant_hits = sum(1 for p in ASSISTANT_PHRASES if p in text_lower)
    assistant_leak = min(assistant_hits / 3, 1.0)  # 3+ = 100%
    
    for phrase in ASSISTANT_PHRASES:
        if phrase in text_lower:
            flags.append(f'"{phrase}" (assistant)')
    
    # Confabulation detection
    for marker in CONFABULATION_MARKERS:
        if marker in text_lower:
            flags.append(f"Confabulation: '{marker}'")
    
    # Third person drift
    for marker in THIRD_PERSON_MARKERS:
        if marker in text_lower:
            flags.append(f"3rd person: '{marker}'")
    
    # Verbosity (tokens vs expected)
    token_count = len(response.split())  # Rough
    verbosity = min(token_count / expected_length, 2.0) / 2.0
    
    # Voice match (inverse of problems)
    problems = assistant_leak + len([f for f in flags if "3rd person" in f]) * 0.2
    voice_match = max(0, 1.0 - problems)
    
    return ResponseAnalysis(
        assistant_leak_score=assistant_leak,
        verbosity_score=verbosity,
        voice_match_score=voice_match,
        flags=flags,
    )
```

## Mock Implementation

```python
# mock_memory.py

import json
from typing import List, Dict, Optional
from dataclasses import dataclass

@dataclass
class MockNode:
    id: str
    type: str
    content: str
    tags: List[str]
    weight: float = 1.0

class CannedMemoryMatrix:
    """Deterministic memory retrieval from JSON snapshot."""
    
    def __init__(self, snapshot_path: str):
        with open(snapshot_path) as f:
            data = json.load(f)
        
        self.nodes = {n["id"]: MockNode(**n) for n in data["nodes"]}
        self.edges = data["edges"]
        self.cache = data.get("retrieval_cache", {})
    
    def search(self, query: str, k: int = 5) -> List[MockNode]:
        """Keyword match against cache, fallback to tag search."""
        query_lower = query.lower()
        
        # Check cache first
        if query_lower in self.cache:
            ids = self.cache[query_lower][:k]
            return [self.nodes[id] for id in ids if id in self.nodes]
        
        # Fallback: match against tags
        matches = []
        for node in self.nodes.values():
            if any(query_lower in tag for tag in node.tags):
                matches.append(node)
            elif query_lower in node.content.lower():
                matches.append(node)
        
        # Sort by weight, return top k
        matches.sort(key=lambda n: n.weight, reverse=True)
        return matches[:k]
    
    def get_context(self, node_id: str, depth: int = 1) -> List[MockNode]:
        """Get connected nodes."""
        connected = set()
        for edge in self.edges:
            if edge["from"] == node_id:
                connected.add(edge["to"])
            elif edge["to"] == node_id:
                connected.add(edge["from"])
        
        return [self.nodes[id] for id in connected if id in self.nodes]
    
    def format_for_prompt(self, nodes: List[MockNode]) -> str:
        """Format retrieved memories for injection."""
        if not nodes:
            return ""
        
        lines = ["## Relevant Memories", ""]
        for node in nodes:
            lines.append(f"- [{node.type}] {node.content}")
        
        return "\n".join(lines)
```
