```arduino
#include <LiquidCrystal.h>
#include <Servo.h>

// Pin definitions
const int WIND_BUTTON = 2;
const int FOG_BUTTON = 3;
const int MIST_BUTTON = 4;
const int GEL_BUTTON = 5;
const int SCENE_BUTTON = 6;

const int WIND_PIN = 9;    // PWM pin for fan
const int FOG_PIN = 10;    // LED indicator for fog
const int MIST_PIN = 11;   // LED indicator for mist
const int GEL_SERVO = 12;  // Servo for gel blaster simulation

// LCD pins
const int RS = A0, EN = A1, D4 = A2, D5 = A3, D6 = A4, D7 = A5;
LiquidCrystal lcd(RS, EN, D4, D5, D6, D7);

Servo gelServo;
int currentScene = 0;
const int TOTAL_SCENES = 5;
String scenes[] = {"LA Downtown", "Venice Beach", "Hollywood", "Griffith Park", "Sunset Blvd"};

bool windActive = false;
bool fogActive = false;
bool mistActive = false;
bool gelActive = false;

void setup() {
  // Initialize buttons with pull-up resistors
  pinMode(WIND_BUTTON, INPUT_PULLUP);
  pinMode(FOG_BUTTON, INPUT_PULLUP);
  pinMode(MIST_BUTTON, INPUT_PULLUP);
  pinMode(GEL_BUTTON, INPUT_PULLUP);
  pinMode(SCENE_BUTTON, INPUT_PULLUP);
  
  // Initialize outputs
  pinMode(WIND_PIN, OUTPUT);
  pinMode(FOG_PIN, OUTPUT);
  pinMode(MIST_PIN, OUTPUT);
  
  // Setup servo
  gelServo.attach(GEL_SERVO);
  gelServo.write(0);
  
  // Setup LCD
  lcd.begin(16, 2);
  updateDisplay();
  
  Serial.begin(9600);
  Serial.println("Interactive Treadmill Prototype");
  Serial.println("-------------------------------");
}

void loop() {
  // Check button inputs (active LOW due to pull-up)
  if (digitalRead(WIND_BUTTON) == LOW) {
    windActive = !windActive;
    digitalWrite(WIND_PIN, windActive ? HIGH : LOW);
    Serial.println(windActive ? "Wind ON" : "Wind OFF");
    delay(200); // Simple debounce
  }
  
  if (digitalRead(FOG_BUTTON) == LOW) {
    fogActive = !fogActive;
    digitalWrite(FOG_PIN, fogActive ? HIGH : LOW);
    Serial.println(fogActive ? "Fog ON" : "Fog OFF");
    delay(200);
  }
  
  if (digitalRead(MIST_BUTTON) == LOW) {
    mistActive = !mistActive;
    digitalWrite(MIST_PIN, mistActive ? HIGH : LOW);
    Serial.println(mistActive ? "Mist ON" : "Mist OFF");
    delay(200);
  }
  
  if (digitalRead(GEL_BUTTON) == LOW) {
    // Activate gel blaster briefly
    gelActive = true;
    gelServo.write(90); // Move servo to simulate gel blast
    Serial.println("Gel Blast!");
    delay(500);
    gelServo.write(0);  // Return to starting position
    gelActive = false;
    delay(200);
  }
  
  if (digitalRead(SCENE_BUTTON) == LOW) {
    // Change scene
    currentScene = (currentScene + 1) % TOTAL_SCENES;
    updateDisplay();
    Serial.println("Scene changed to: " + scenes[currentScene]);
    delay(200);
  }
}

void updateDisplay() {
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("Scene:");
  lcd.setCursor(0, 1);
  lcd.print(scenes[currentScene]);
}
```
