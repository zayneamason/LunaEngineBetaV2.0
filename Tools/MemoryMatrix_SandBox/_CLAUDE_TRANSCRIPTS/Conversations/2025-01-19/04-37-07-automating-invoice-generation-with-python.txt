```python
# Option 1: Using win32com (Windows only)
def convert_to_pdf(excel_path, pdf_path):
    import win32com.client
    excel = win32com.client.Dispatch("Excel.Application")
    sheets = excel.Workbooks.Open(excel_path)
    work_sheets = sheets.Worksheets[0]
    work_sheets.ExportAsFixedFormat(0, pdf_path)
    sheets.Close(True)

# Option 2: Using Numbers on macOS (requires applescript)
def convert_to_pdf_macos(excel_path, pdf_path):
    import subprocess
    script = f'''
    tell application "Numbers"
        open "{excel_path}"
        export front document to "{pdf_path}" as PDF
        close front document saving no
    end tell
    '''
    subprocess.run(['osascript', '-e', script])
```
