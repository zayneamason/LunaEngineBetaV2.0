### Batch Processing Logic

```javascript
// batchProcessor.js

async function processBatch(imagePaths, options = {}) {
  const { 
    concurrency = 3,        // Parallel API calls
    delayMs = 500,          // Delay between batches
    skipExisting = true     // Skip if file_hash already in DB
  } = options;
  
  const results = [];
  const batches = chunk(imagePaths, concurrency);
  
  for (const batch of batches) {
    const batchResults = await Promise.all(
      batch.map(async (path) => {
        // Check if already scanned
        if (skipExisting) {
          const hash = await getFileHash(path);
          const existing = await db.getByHash(hash);
          if (existing) return { path, skipped: true, existing };
        }
        
        // Analyze with Claude Vision
        const analysis = await imageAnalyzer.analyze(path);
        
        // Save to DB
        await db.insertImage({ path, ...analysis });
        
        return { path, analysis };
      })
    );
    
    results.push(...batchResults);
    await sleep(delayMs);
  }
  
  return results;
}
```
