```python
def select_tier(classification: ClassificationResult) -> int:
    """
    Score 0.0-1.0 → Tier 0-4
    
    Thresholds:
    - 0.0-0.2: T0/T1 (reflex)
    - 0.2-0.4: T1 (simple chat)
    - 0.4-0.6: T2 (warm voice)
    - 0.6-0.8: T3 (thinker)
    - 0.8-1.0: T4 (full Luna)
    """
    score = compute_routing_score(classification)
    
    if score < 0.2:
        return 1 if classification.requires_memory else 0
    elif score < 0.4:
        return 1
    elif score < 0.6:
        return 2
    elif score < 0.8:
        return 3
    else:
        return 4
```

#### M7: Basic Orchestrator
```python
# eclissi/orchestrator.py
class Orchestrator:
    def __init__(self):
        self.classifier = Classifier()
        self.lifecycle = LifecycleManager()
        self.context = ContextBuilder()
        
    async def process(self, user_input: str) -> Response:
        # 1. Classify
        classification = await self.classifier.classify(user_input)
        
        # 2. Select tier
        tier = select_tier(classification)
        
        # 3. Build context
        context = self.context.build_for_tier(tier, user_input, classification)
        
        # 4. Acquire model
        model = MODEL_REGISTRY[tier_to_model(tier)]
        handle = await self.lifecycle.acquire(model.id)
        
        # 5. Generate
        response = await handle.generate(context)
        
        # 6. Release
        await self.lifecycle.release(model.id)
        
        return Response(text=response, tier=tier, classification=classification)
```

**Deliverable:** End-to-end query → response with tier routing
**Test:** Multiple queries, verify correct tier selection

Six fields from a 3B model in ~50ms? That's ambitious. Phi-3 is good but not magic.

- Have a fallback: if JSON parse fails, default to T2 (safe middle ground)

def simple_lowpass(audio: np.ndarray, cutoff_ratio: float = 0.5, passes: int = 1) -> np.ndarray:
    """
    Simple moving-average lowpass filter (no scipy needed).
    cutoff_ratio: 0.0-1.0, higher = less filtering (more clarity)
    """
    # Window size based on cutoff (larger window = more smoothing)
    window_size = max(3, int((1.0 - cutoff_ratio) * 9) | 1)  # ensure odd, smaller max
    kernel = np.ones(window_size) / window_size
    for _ in range(passes):
        audio = np.convolve(audio, kernel, mode='same')
    return audio
def luna_post_process(audio: np.ndarray, sr: int = 24000,
