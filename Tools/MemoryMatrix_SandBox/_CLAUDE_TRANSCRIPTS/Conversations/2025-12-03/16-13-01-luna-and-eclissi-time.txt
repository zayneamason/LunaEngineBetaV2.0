```python
# apps/voice/luna/eclissi_brain.py
"""
Luna's brain - powered by Eclissi.

Wraps the Eclissi orchestrator for voice integration with
maintenance scheduling, tier routing, and MEMORY RETRIEVAL.
"""
import asyncio
from typing import Optional, Dict, List
from pathlib import Path

# Import Eclissi
import sys
sys.path.insert(0, str(Path.home() / "HeyLuna"))

from src.eclissi.orchestrator import Orchestrator
from src.eclissi.scheduler.maintenance import MaintenanceScheduler

# Memory retrieval
try:
    from src.memory.retrieval.scorer import RetrievalScorer
    from src.memory.retrieval.context_builder import build_context_packet, format_for_injection
    MEMORY_AVAILABLE = True
except ImportError as e:
    print(f"[EclissiBrain] Memory system not available: {e}")
    MEMORY_AVAILABLE = False


class EclissiBrain:
    """
    Luna's brain - now powered by Eclissi.

    Provides:
    - Tier-based routing (T1-T4)
    - Automatic fallback on failures
    - Background maintenance scheduling
    - Metrics for Overwatch UI
    - MEMORY RETRIEVAL (NEW)
    """

    def __init__(self):
        self.eclissi = Orchestrator()
        self.scheduler = MaintenanceScheduler(orchestrator=self.eclissi)
        self._scheduler_task = None
        self._started = False
        
        # Memory system
        if MEMORY_AVAILABLE:
            self._scorer = RetrievalScorer()
            print("[EclissiBrain] Memory system loaded")
        else:
            self._scorer = None

    async def start(self):
        """Start brain services."""
        if self._started:
            return

        self._scheduler_task = asyncio.create_task(
            self.scheduler.run_loop(check_interval=2.0)
        )
        self._started = True
        print("[EclissiBrain] Started")

    async def stop(self):
        """Stop brain services."""
        if self._scheduler_task:
            self._scheduler_task.cancel()
            try:
                await self._scheduler_task
            except asyncio.CancelledError:
                pass
        self._started = False
        print("[EclissiBrain] Stopped")

    def _fetch_memories(self, query: str, budget: str = "balanced") -> List[str]:
        """
        Fetch relevant memories for the query.
        
        Args:
            query: User input to find relevant context for
            budget: Token budget - 'minimal', 'balanced', or 'rich'
            
        Returns:
            List of memory strings ready for context injection
        """
        if not self._scorer:
            return []
            
        try:
            # Score memories against query
            scored = self._scorer.score_memories(query)
            
            if not scored:
                return []
            
            # Build context packet
            packet = build_context_packet(
                query=query,
                scored_memories=scored,
                budget_preset=budget,
                app_context=("voice", "conversation")  # We're in voice app
            )
            
            # Format for injection
            context_str = format_for_injection(packet)
            
            # Log what we fetched
            meta = packet.retrieval_metadata
            print(f"[EclissiBrain] Fetched {meta.get('memories_selected', 0)} memories "
                  f"({packet.budget_used} tokens)")
            
            # Return as list (orchestrator expects List[str])
            return [context_str] if context_str else []
            
        except Exception as e:
            print(f"[EclissiBrain] Memory fetch failed: {e}")
            return []

    def respond(self, user_input: str, memories: List[str] = None) -> str:
        """
        Main entry point for voice app.
        Suspends maintenance, FETCHES MEMORIES, processes query, resumes.

        Args:
            user_input: What the user said
            memories: Optional pre-fetched memories (if None, auto-fetches)

        Returns:
            Luna's response text
        """
        # Suspend background work
        if self._started:
            asyncio.create_task(self.scheduler.suspend())
        self.eclissi.suspend_maintenance()

        try:
            # AUTO-FETCH MEMORIES if not provided
            if memories is None:
                memories = self._fetch_memories(user_input, budget="balanced")
            
            response = self.eclissi.process(user_input, memories=memories)
            return response.text
        finally:
            # Always resume maintenance
            self.eclissi.resume_maintenance()
            if self._started:
                asyncio.create_task(self.scheduler.resume())

    def respond_with_meta(self, user_input: str, memories: List[str] = None) -> dict:
        """
        Respond with full metadata (for Overwatch UI).

        Returns dict with:
            text: Response text
            tier: Which tier was used
            complexity: Classification score
            latency_ms: Total latency
            classifier_ms: Classifier latency
            fallback: Whether fallback was used
            memories_loaded: How many memories were fetched
        """
        if self._started:
            asyncio.create_task(self.scheduler.suspend())
        self.eclissi.suspend_maintenance()

        try:
            # AUTO-FETCH MEMORIES if not provided
            if memories is None:
                memories = self._fetch_memories(user_input, budget="balanced")
            
            response = self.eclissi.process(user_input, memories=memories)
            return {
                "text": response.text,
                "tier": response.tier,
                "complexity": response.classification.complexity,
                "latency_ms": response.latency_ms,
                "classifier_ms": response.classification.latency_ms,
                "fallback": response.fallback_used,
                "parse_success": response.classification.parse_success,
                "memories_loaded": len(memories) if memories else 0
            }
        finally:
            self.eclissi.resume_maintenance()
            if self._started:
                asyncio.create_task(self.scheduler.resume())

    def get_tier_for_display(self) -> int:
        """Get current tier for debug console."""
        metrics = self.eclissi._metrics
        if metrics:
            return metrics[-1].tier
        return 0

    def get_status(self) -> dict:
        """Full status for debug console / Overwatch UI."""
        return {
            "eclissi": {
                "metrics": self.eclissi.get_metrics_summary(),
                "lifecycle": self.eclissi.lifecycle.status(),
                "recent": self.eclissi.get_recent_metrics(5)
            },
            "scheduler": self.scheduler.stats(),
            "started": self._started,
            "memory_available": MEMORY_AVAILABLE
        }

    def set_voice_active(self, active: bool):
        """
        Voice pipeline controls maintenance permission.

        Call with active=True when user is speaking or Luna is responding.
        Call with active=False during true idle periods.
        """
        self.scheduler.set_maintenance_allowed(not active)

    def preload_models(self):
        """Preload hot models (T0, T1) for faster first response."""
        self.eclissi.lifecycle.preload_hot_models()

    def reset_session(self):
        """Reset session state (new conversation)."""
        self.eclissi.reset_session()


# Convenience function for quick testing
def test_brain():

    asyncio.run(run())


if __name__ == "__main__":
```

```python
    for pattern, score, emo in COMPLEX_TRIGGERS:

            complexity = max(complexity or 0, score)
            emotion = emo
    
    # If we found triggers but not a simple pattern, return partial result
    if needs_memory or complexity:
        # Extract any obvious topics from the text
        topics = _extract_topics(text_lower)
        
        return ClassificationResult(
            complexity=complexity or 0.5,
            emotion=emotion,
            needs_memory=needs_memory,
            topics=topics,
            memory_hints=memory_hints,
            fast_path=True,
            parse_success=True,
        )
    
    return None  # No fast path match, use LLM


def _extract_topics(text: str) -> List[str]:
    """Extract obvious topics from text."""
    topics = []
    
    # Technical topics
    if re.search(r'\b(code|programming|python|javascript|api)\b', text, re.I):
        topics.append("technical")
    if re.search(r'\b(luna|eclissi|voice|memory|virtue)\b', text, re.I):
        topics.append("luna_system")
    if re.search(r'\b(bug|error|fix|debug|broken)\b', text, re.I):
        topics.append("debugging")
    if re.search(r'\b(build|create|make|design)\b', text, re.I):
        topics.append("development")
        
    # Emotional/personal topics
    if re.search(r'\b(feel|feeling|felt|emotion)\b', text, re.I):
        topics.append("emotional")
    if re.search(r'\b(help|stuck|confused|frustrated)\b', text, re.I):
        topics.append("assistance")
        
    return topics if topics else ["general"]


class Classifier:
    """Phi-3 Oracle classifier using MLX."""

    def __init__(self):
        config = get_config()
        self.model_id = config.mlx_models["T0"]
        self.timeout_ms = config.classifier_timeout_ms
        self._model = None
        self._tokenizer = None
        self._loaded = False

    def _ensure_loaded(self):
        """Lazy load the model."""
        if self._loaded:
            return

        from mlx_lm import load
        print(f"[Classifier] Loading {self.model_id}...")
        self._model, self._tokenizer = load(self.model_id)
        self._loaded = True
        print(f"[Classifier] Model loaded")

    def classify(self, text: str, session_turn: int = 0) -> ClassificationResult:
        """
        Full Oracle classification.

        Args:
            text: User input
            session_turn: Current turn number (for bias)

        Returns:
            ClassificationResult with complexity, topics, emotion, memory hints
        """
        start = time.perf_counter()

        # FAST PATH: Check patterns first
        result = _fast_path_classify(text)
        if result is not None:
            # Apply session turn bias
            result = self._apply_turn_bias(result, session_turn)
            result.latency_ms = (time.perf_counter() - start) * 1000
            print(f"[Oracle] Fast: c={result.complexity:.2f} e={result.emotion} "
                  f"mem={result.needs_memory} t={result.topics} ({result.latency_ms:.1f}ms)")
            return result

        # SLOW PATH: Full LLM Oracle
        try:
            self._ensure_loaded()

            prompt = ORACLE_PROMPT + text[:300]  # Truncate long inputs

            # Apply chat template if available
            if hasattr(self._tokenizer, 'apply_chat_template') and self._tokenizer.chat_template:
                messages = [{"role": "user", "content": prompt}]
                prompt = self._tokenizer.apply_chat_template(
                    messages,
                    tokenize=False,
                    add_generation_prompt=True
                )

            from mlx_lm import generate
            raw = generate(
                self._model,
                self._tokenizer,
                prompt=prompt,
                max_tokens=100,  # More tokens for full JSON
                verbose=False
            )

            latency = (time.perf_counter() - start) * 1000
            result = self._parse_oracle_response(raw, text, latency)
            result = self._apply_turn_bias(result, session_turn)
            
            print(f"[Oracle] LLM: c={result.complexity:.2f} e={result.emotion} "
                  f"mem={result.needs_memory} t={result.topics} ({result.latency_ms:.1f}ms)")
            return result

        except Exception as e:
            latency = (time.perf_counter() - start) * 1000
            print(f"[Oracle] Error: {e}")
            # Fallback with topic extraction
            return ClassificationResult(
                complexity=0.5,
                emotion="neutral",
                needs_memory=False,
                topics=_extract_topics(text),
                memory_hints=[],
                raw_response=f"ERROR: {e}",
                parse_success=False,
                latency_ms=latency,
            )

    def _parse_oracle_response(self, raw: str, original_text: str, latency_ms: float) -> ClassificationResult:
        """Parse full Oracle JSON response."""
        try:
            match = re.search(r'\{[^}]+\}', raw, re.DOTALL)
            if match:
                data = json.loads(match.group())
                
                return ClassificationResult(
                    complexity=float(data.get("complexity", 0.5)),
                    emotion=data.get("emotion", "neutral"),
                    needs_memory=bool(data.get("needs_memory", False)),
                    topics=data.get("topics", []) or _extract_topics(original_text),
                    memory_hints=data.get("memory_hints", []),
                    raw_response=raw,
                    parse_success=True,
                    latency_ms=latency_ms,
                )
        except Exception as e:
            print(f"[Oracle] Parse failed: {e}")

        # Parse failed - extract what we can
        return ClassificationResult(
            complexity=0.5,
            emotion="neutral",
            needs_memory=False,
            topics=_extract_topics(original_text),
            memory_hints=[],
            raw_response=raw,
            parse_success=False,
            latency_ms=latency_ms,
        )

    def _apply_turn_bias(self, result: ClassificationResult, session_turn: int) -> ClassificationResult:
        """Apply session turn bias to complexity."""
        if session_turn >= 5:
            result.complexity = min(1.0, result.complexity + 0.2)
            result.needs_memory = True  # Deep in convo, probably need context
        elif session_turn >= 3:
            result.complexity = min(1.0, result.complexity + 0.1)
        
        return result
```
