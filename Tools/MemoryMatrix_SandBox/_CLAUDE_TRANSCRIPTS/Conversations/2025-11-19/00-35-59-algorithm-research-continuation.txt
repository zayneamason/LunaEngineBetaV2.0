const sectionVariationCache = new Map();

function getCachedVariation(sectionId, seed, detailLevel) {
  const key = hash(`${sectionId}:${seed}:${detailLevel}`);
  
  if (sectionVariationCache.has(key)) {
    return sectionVariationCache.get(key); // O(1) retrieval
  }
  
  // Generate on miss
  const variation = generateVariation(sectionId, seed, detailLevel);
  sectionVariationCache.set(key, variation);
  return variation;
}

- Different seed = different cache key
