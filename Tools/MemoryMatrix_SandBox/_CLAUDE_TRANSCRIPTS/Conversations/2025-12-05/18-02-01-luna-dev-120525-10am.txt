## Scoring - Flexible Approach

Instead of hardcoding Phi-3, abstract to a `MemoryEvaluator` interface:

```python
class MemoryEvaluator(Protocol):
    def score(self, turns: List[Turn]) -> float:
        """Return 0.0-1.0 save-worthiness score"""
        ...

class HeuristicEvaluator:
    """Fast, always available, no API cost"""
    def score(self, turns: List[Turn]) -> float:
        score = 0.0
        
        # Length signals substance
        total_words = sum(len(t.luna.split()) for t in turns)
        if total_words > 200: score += 0.2
        if total_words > 500: score += 0.1
        
        # Emotional content
        non_neutral = sum(1 for t in turns if t.oracle.emotion != "neutral")
        if non_neutral > 0: score += 0.2
        if non_neutral > 2: score += 0.1
        
        # Explicit cues
        if any(cue in t.user.lower() or cue in t.luna.lower() 
               for t in turns for cue in EXPLICIT_CUES):
            score += 0.4
        
        # Implicit cues
        if any(cue in t.user.lower() or cue in t.luna.lower()
               for t in turns for cue in IMPLICIT_CUES):
            score += 0.2
        
        # Memory was needed (Oracle said so)
        if any(t.oracle.needs_memory for t in turns):
            score += 0.1
        
        # High complexity conversation
        avg_complexity = sum(t.oracle.complexity for t in turns) / len(turns)
        if avg_complexity > 0.6: score += 0.15
        
        return min(score, 1.0)

class LLMEvaluator:
    """More nuanced, costs tokens/latency"""
    def __init__(self, provider: str = "local"):
        # "local" = Phi-3/Ollama, "haiku" = Claude, "groq" = fast cloud
        self.provider = provider
    
    def score(self, turns: List[Turn]) -> float:
        prompt = f"""Rate this conversation segment 0-10 for long-term memory value.
Consider: insights learned, decisions made, emotional significance, project progress.
Just respond with a number.

{self._format_turns(turns)}"""
        
        # Route to provider...
        raw_score = self._call_llm(prompt)
        return float(raw_score) / 10.0
```

**Scoring strategy:**

```python
class CompositeEvaluator:
    def __init__(self):
        self.heuristic = HeuristicEvaluator()
        self.llm = LLMEvaluator(provider="local")  # configurable
        self.llm_threshold = 0.4  # only call LLM if heuristic is ambiguous
    
    def score(self, turns: List[Turn]) -> float:
        h_score = self.heuristic.score(turns)
        
        # Clear signal - skip LLM
        if h_score >= 0.7:  # obviously save
            return h_score
        if h_score <= 0.2:  # obviously skip
            return h_score
        
        # Ambiguous - ask LLM
        l_score = self.llm.score(turns)
        
        # Blend (LLM weighted higher when used)
        return 0.3 * h_score + 0.7 * l_score
```
