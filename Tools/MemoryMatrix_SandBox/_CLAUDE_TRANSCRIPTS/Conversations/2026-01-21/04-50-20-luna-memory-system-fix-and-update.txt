## Simple Implementation

```python
from collections import deque

class ConversationRing:
    """
    Guaranteed working memory. Never displaced by retrieval.
    """
    def __init__(self, max_turns: int = 6):
        self._buffer = deque(maxlen=max_turns)
    
    def add(self, role: str, content: str):
        self._buffer.append({"role": role, "content": content})
    
    def get_history(self) -> list:
        return list(self._buffer)
    
    def contains_topic(self, topic: str) -> bool:
        """Check if topic was discussed recently."""
        text = " ".join(m["content"].lower() for m in self._buffer)
        return topic.lower() in text
    
    def format_for_prompt(self) -> str:
        lines = []
        for i, msg in enumerate(self._buffer):
            role = "Ahab" if msg["role"] == "user" else "Luna"
            lines.append(f"[{i+1}] {role}: {msg['content']}")
        return "\n".join(lines)
```

```python
class ContextPipeline:
    """
    Unified context building for ALL inference paths.
    
    The inference engine (Claude/MLX) is just a completion backend.
    Context is built identically regardless of routing decision.
    """
    
    def __init__(self, ring: ConversationRing, entity_context: EntityContext, matrix: MemoryMatrix):
        self._ring = ring
        self._entity_context = entity_context
        self._matrix = matrix
    
    async def build(self, message: str) -> ContextPacket:
        """
        Build complete context packet for any inference path.
        """
        # 1. Add to ring (guaranteed history)
        self._ring.add_user(message)
        
        # 2. Entity detection
        entities = await self._entity_context.detect_and_load(message)
        
        # 3. Self-route decision for memory
        need_retrieval = not self._ring.contains_any([e.name for e in entities])
        memories = await self._matrix.search(message) if need_retrieval else []
        
        # 4. Build framed context
        framed = self._build_framed(
            ring_history=self._ring.format_for_prompt(),
            entities=entities,
            memories=memories,
        )
        
        return ContextPacket(
            system_prompt=framed,
            messages=self._ring.get_as_dicts(),
            entities=entities,
        )


class Director:
    async def process(self, message: str) -> str:
        # UNIFIED: Same context regardless of route
        context = await self._context_pipeline.build(message)
        
        # Routing decision (complexity, token count, etc.)
        if self._should_delegate(message, context):
            response = await self._claude_client.complete(context)
        else:
            response = await self._local_model.complete(context)
        
        # Update ring with response
        self._context_pipeline.ring.add_assistant(response)
        
        return response
```
