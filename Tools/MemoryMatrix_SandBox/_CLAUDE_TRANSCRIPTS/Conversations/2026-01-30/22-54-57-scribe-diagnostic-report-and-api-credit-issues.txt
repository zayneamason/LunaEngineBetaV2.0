# Scribe Fallback Chain Architecture

## Problem Statement

Current state:
- Scribe backend = `haiku` (Claude API)
- Claude API out of credits → silent failure
- `_extract_local()` exists but falls through to empty result
- No error surfacing, no retry, no fallback cascade

Result: Extraction stops silently. Raw turns still pollute memory_nodes.

---

## Design Goals

1. **Graceful degradation**: Always extract *something*, even if lower quality
2. **Sovereignty-first**: Local should be the fallback floor, not the fallthrough
3. **Observable failures**: Know when backends fail and why
4. **Zero config hot-swap**: Switch backends without restart
5. **Cost-aware**: Track usage, respect rate limits

---

## Architecture

### Fallback Chain Definition

```python
@dataclass
class ExtractionBackend:
    """A single backend in the fallback chain."""
    name: str                    # "anthropic", "groq", "local"
    provider: str                # Provider ID or "local"
    model: str                   # Model identifier
    priority: int                # Lower = preferred
    is_available: Callable[[], bool]  # Availability check
    quality: str                 # "excellent", "good", "basic"
    cost_per_1k: float          # Input token cost (for tracking)
    
@dataclass
class FallbackChain:
    """Ordered list of extraction backends."""
    backends: list[ExtractionBackend]
    current_index: int = 0
    failure_counts: dict[str, int] = field(default_factory=dict)
    
    def get_active(self) -> Optional[ExtractionBackend]:
        """Get current active backend, respecting availability."""
        for backend in sorted(self.backends, key=lambda b: b.priority):
            if backend.is_available():
                return backend
        return None
    
    def mark_failure(self, backend_name: str, error: str) -> None:
        """Record a failure, potentially triggering fallback."""
        self.failure_counts[backend_name] = self.failure_counts.get(backend_name, 0) + 1
    
    def reset_failures(self, backend_name: str) -> None:
        """Reset failure count after success."""
        self.failure_counts[backend_name] = 0
```

## ExtractionRouter Implementation

```python
@dataclass
class BackendError:
    """Record of a backend failure."""
    backend: str
    error_type: str  # "rate_limit", "auth", "timeout", "parse", "unknown"
    message: str
    timestamp: float
    
@dataclass  
class ExtractionResult:
    """Full result with diagnostics."""
    output: ExtractionOutput
    entity_updates: list[EntityUpdate]
    backend_used: str
    fallback_triggered: bool
    errors: list[BackendError]
    extraction_time_ms: int


class ExtractionRouter:
    """Routes extraction to available backends with fallback."""
    
    def __init__(self, chain: FallbackChain, engine: "LunaEngine"):
        self.chain = chain
        self.engine = engine
        self._providers: dict[str, LLMProvider] = {}
        
    async def extract(
        self, 
        text: str, 
        source_id: str
    ) -> ExtractionResult:
        """
        Attempt extraction with fallback chain.
        
        Tries each backend in priority order until one succeeds.
        """
        errors: list[BackendError] = []
        fallback_triggered = False
        
        for backend in self._iter_available_backends():
            try:
                output, updates = await self._try_backend(backend, text, source_id)
                
                if not output.is_empty() or updates:
                    # Success - reset failure count
                    self.chain.reset_failures(backend.name)
                    
                    return ExtractionResult(
                        output=output,
                        entity_updates=updates,
                        backend_used=backend.name,
                        fallback_triggered=fallback_triggered,
                        errors=errors,
                        extraction_time_ms=output.extraction_time_ms,
                    )
                    
            except RateLimitError as e:
                errors.append(BackendError(
                    backend=backend.name,
                    error_type="rate_limit",
                    message=str(e),
                    timestamp=time.time(),
                ))
                self.chain.mark_failure(backend.name, "rate_limit")
                fallback_triggered = True
                
            except AuthenticationError as e:
                errors.append(BackendError(
                    backend=backend.name,
                    error_type="auth",
                    message=str(e),
                    timestamp=time.time(),
                ))
                self.chain.mark_failure(backend.name, "auth")
                fallback_triggered = True
                
            except Exception as e:
                errors.append(BackendError(
                    backend=backend.name,
                    error_type="unknown",
                    message=str(e),
                    timestamp=time.time(),
                ))
                self.chain.mark_failure(backend.name, "unknown")
                fallback_triggered = True
        
        # All backends failed
        logger.error(f"All extraction backends failed: {[e.backend for e in errors]}")
        
        return ExtractionResult(
            output=ExtractionOutput(),
            entity_updates=[],
            backend_used="none",
            fallback_triggered=True,
            errors=errors,
            extraction_time_ms=0,
        )
    
    def _iter_available_backends(self) -> Iterator[ExtractionBackend]:
        """Yield backends in priority order, skipping unavailable."""
        for backend in sorted(self.chain.backends, key=lambda b: b.priority):
            if backend.is_available():
                yield backend
                
    async def _try_backend(
        self, 
        backend: ExtractionBackend,
        text: str,
        source_id: str,
    ) -> tuple[ExtractionOutput, list[EntityUpdate]]:
        """Attempt extraction with a specific backend."""
        
        if backend.provider == "local":
            return await self._extract_local(text, source_id)
        
        if backend.provider == "anthropic":
            return await self._extract_anthropic(backend.model, text, source_id)
            
        if backend.provider == "groq":
            return await self._extract_groq(backend.model, text, source_id)
            
        raise ValueError(f"Unknown provider: {backend.provider}")
```

## The Question: Who Should Own Local Inference?

### Option A: Shared Instance (Current + Formalized)

One `LocalInference` instance, but owned by the **Engine**, not Director:

```
┌─────────────────────────────────────────────┐
│                  Engine                      │
│                                             │
│  ┌─────────────────────────────────────┐   │
│  │         LocalInference              │   │
│  │  (single Qwen model in memory)      │   │
│  └──────────────┬──────────────────────┘   │
│                 │                           │
│      ┌──────────┼──────────┐               │
│      ▼          ▼          ▼               │
│  ┌───────┐ ┌────────┐ ┌──────────┐        │
│  │Director│ │ Scribe │ │Librarian │        │
│  └───────┘ └────────┘ └──────────┘        │
└─────────────────────────────────────────────┘
```

**Pros:**
- One model loaded = ~3GB VRAM once
- Clear ownership (Engine provides, actors consume)
- No duplication

**Cons:**
- Contention if Director and Scribe both need it simultaneously
- Single point of failure
- Can't tune model differently for different tasks

---

### Option B: Scribe Gets Its Own Instance

Each actor that needs local inference gets its own:

```
┌─────────────────────────────────────────────┐
│                  Engine                      │
│                                             │
│  ┌───────────────────┐ ┌─────────────────┐ │
│  │     Director      │ │     Scribe      │ │
│  │ ┌───────────────┐ │ │ ┌─────────────┐ │ │
│  │ │LocalInference │ │ │ │LocalInfer.  │ │ │
│  │ │(conversation) │ │ │ │(extraction) │ │ │
│  │ └───────────────┘ │ │ └─────────────┘ │ │
│  └───────────────────┘ └─────────────────┘ │
└─────────────────────────────────────────────┘
```

**Pros:**
- No contention
- Could use different models (Director: fast 3B, Scribe: maybe 7B for better extraction)
- Could use different LoRA adapters
- Clean separation of concerns

**Cons:**
- 2x memory usage (~6GB VRAM for two models)
- More complexity
- Startup time doubled

One loaded model, but actors get their own "view" with different generation configs:

```python
class LocalInferencePool:

    def __init__(self):
```

**Pros:**
- Single model in memory
- Different temperature/prompts per actor
- Explicit dependency injection

**Cons:**
- Still can't use different *models* per actor

## My Recommendation: Option A (Formalized)

For Luna's current scale and M1 hardware constraints:

1. **Move LocalInference to Engine level** - Engine owns it, actors request it
2. **Explicit injection** - Pass reference in actor constructors, not runtime discovery
3. **Async queue for contention** - If both need it, queue requests

```python
# Engine initialization
class LunaEngine:
    def __init__(self, ...):
        self._local_inference = LocalInference()
        
        # Inject into actors that need it
        self._scribe = ScribeActor(
            local_inference=self._local_inference,
            ...
        )
        self._director = DirectorActor(
            local_inference=self._local_inference,
            ...
        )
```

```python
# Scribe uses it cleanly
class ScribeActor(Actor):
    def __init__(self, local_inference: LocalInference, ...):
        self._local = local_inference  # Injected, not discovered
        
    async def _extract_local(self, text: str, source_id: str):
```

This removes the "reach through engine to borrow Director's model" smell while keeping memory usage sane.

## The Contention Problem (Revisited)

Now the con "contention if Director and Scribe both need it simultaneously" becomes more specific:

```
Timeline:
─────────────────────────────────────────────────────────────────►
     │                    │                    │
     ▼                    ▼                    ▼

  Luna LoRA]          Scribe LoRA]         Luna LoRA again]

     └─── SWAP ───────────┴─── SWAP ──────────┘
```

1. Director loads Luna LoRA, starts generating
2. Scribe needs extraction, swaps to Scribe LoRA **mid-generation**
3. Director's output gets corrupted

**This is a real race condition.**

## Solution: LoRA-Aware Request Queue

```python
@dataclass
class InferenceRequest:
    """A queued inference request with LoRA context."""
    actor: str
    lora_id: str  # "luna", "scribe", "librarian", None
    prompt: str
    config: GenerationConfig
    future: asyncio.Future  # Caller awaits this


class LoRAInferencePool:
    """
    Shared base model with hot-swappable LoRA adapters.
    
    Serializes requests to prevent LoRA conflicts.
    Batches requests with same LoRA for efficiency.
    """
    
    def __init__(self):
        self._model = None
        self._tokenizer = None
        self._current_lora: Optional[str] = None
        self._lora_cache: dict[str, Any] = {}  # Preloaded adapter weights
        self._request_queue: asyncio.Queue[InferenceRequest] = asyncio.Queue()
        self._processing = False
        
    async def load_base_model(self, model_id: str):
        """Load base model without any LoRA."""
        self._model, self._tokenizer = load(model_id, adapter_path=None)
        
    def register_lora(self, lora_id: str, adapter_path: str):
        """Preload a LoRA adapter into cache."""
        weights = mx.load(f"{adapter_path}/adapters.safetensors")
        self._lora_cache[lora_id] = weights
        logger.info(f"Registered LoRA: {lora_id}")
        
    async def generate(
        self,
        actor: str,
        lora_id: Optional[str],
        prompt: str,
        config: GenerationConfig,
    ) -> str:
        """
        Queue a generation request.
        
        Caller awaits result. Pool handles LoRA swapping.
        """
        future = asyncio.Future()
        request = InferenceRequest(
            actor=actor,
            lora_id=lora_id,
            prompt=prompt,
            config=config,
            future=future,
        )
        await self._request_queue.put(request)
        
        # Ensure processor is running
        if not self._processing:
            asyncio.create_task(self._process_queue())
            
        return await future
        
    async def _process_queue(self):
        """Process requests, batching same-LoRA requests."""
        self._processing = True
        
        try:
            while True:
                # Get next request
                request = await asyncio.wait_for(
                    self._request_queue.get(),
                    timeout=1.0
                )
                
                # Swap LoRA if needed
                if request.lora_id != self._current_lora:
                    await self._swap_lora(request.lora_id)
                
                # Generate
                try:
                    result = await self._generate_internal(
                        request.prompt,
                        request.config,
                    )
                    request.future.set_result(result)
                except Exception as e:
                    request.future.set_exception(e)
                    
        except asyncio.TimeoutError:
            pass  # Queue empty, stop processing
        finally:
            self._processing = False
            
    async def _swap_lora(self, lora_id: Optional[str]):
        """Hot-swap the active LoRA adapter."""
        if lora_id is None:
            # Clear LoRA, use base model
            # (MLX may require model reload here - need to verify)
            pass
        elif lora_id in self._lora_cache:
            weights = self._lora_cache[lora_id]
            apply_lora_layers(self._model, weights)
            logger.debug(f"Swapped to LoRA: {lora_id}")
        else:
            raise ValueError(f"Unknown LoRA: {lora_id}")
            
        self._current_lora = lora_id
```

## Summary: MLX LoRA Hot-Swapping Analysis

### Key Findings

**Yes, MLX supports hot-swapping LoRA adapters.** Here's how it works:

```
┌─────────────────────────────────────────────────────────────────┐
│                    LoRALinear Layer                             │
│                                                                 │
│   Input (x) ──┬──► [Base Linear] ──► y (base output)           │
│               │                           │                     │
│               └──► [x @ lora_a @ lora_b] ► z (LoRA delta)      │
│                                           │                     │
│                           Output = y + (scale * z)              │
└─────────────────────────────────────────────────────────────────┘
```

The base weights are **preserved inside `lora.linear`**. The LoRA delta (`lora_a`, `lora_b`) is **additive**. To swap adapters, you just reassign these tensors.

```python
class MultiLoRAPool:
```

**Proposed Data Structures**

```python
@dataclass
class PersonaConfig:
    """A complete persona configuration bundle."""
    
    name: str                          # "luna", "scribe", "librarian"
    
    # Model selection
    model_id: str                      # "local", "claude-sonnet", "groq-llama"
    model_fallback: Optional[str]      # Fallback if primary unavailable
    
    # LoRA adapter (optional)
    adapter_name: Optional[str]        # "luna_lora", "scribe_lora", None
    adapter_path: Optional[str]        # Path to adapter files
    
    # Prompting
    system_prompt_path: str            # Path to system prompt template
    prompt_variables: Dict[str, Any]   # Variables to inject into prompt
    
    # Personality (for context-aware personas)
    personality_params: Optional[Dict[str, float]]  # warmth, technical, etc.
    context_state: Optional[str]       # "professional", "personal", etc.
    
    # Capabilities
    tools: List[str]                   # Available tool names
    memory_scope: List[str]            # Memory contexts to access
    output_format: str                 # "text", "json", "structured"
    
    # Constraints
    max_tokens: int = 512
    temperature: float = 0.7


class PersonaRegistry:
    """Central registry for all persona configurations."""
    
    def __init__(self, config_dir: Path):
        self._personas: Dict[str, PersonaConfig] = {}
        self._variants: Dict[str, Dict[str, str]] = {}  # persona -> {context: variant}
        self._load_configs(config_dir)
    
    def get(self, name: str, context: Optional[str] = None) -> PersonaConfig:
        """
        Get persona config, optionally resolved for context.
        
        Examples:
            registry.get("luna")  # Base Luna
            registry.get("luna", context="deep_work")  # Luna focus variant
            registry.get("scribe")  # Scribe (no context variants)
        """
        if context and name in self._variants:
            variant_name = self._variants[name].get(context, name)
            return self._personas[variant_name]
        return self._personas[name]
    
    def register(self, config: PersonaConfig) -> None:
        """Register a persona configuration."""
        self._personas[config.name] = config
    
    def register_variant(self, base: str, context: str, variant: str) -> None:
        """Register a context variant for a persona."""
        if base not in self._variants:
            self._variants[base] = {}
        self._variants[base][context] = variant


class PersonaRouter:
    """Routes requests to appropriate persona configurations."""
    
    def __init__(
        self,
        registry: PersonaRegistry,
        model_pool: ModelPool,       # Existing model routing
        lora_pool: MultiLoRAPool,    # From our spec
        prompt_cache: PromptCache,
    ):
        self._registry = registry
        self._models = model_pool
        self._loras = lora_pool
        self._prompts = prompt_cache
    
    async def resolve(
        self,
        persona: str,
        context: Optional[str] = None,
    ) -> "ResolvedPersona":
        """
        Resolve persona name + context into a ready-to-use generation pipeline.
        
        Returns a ResolvedPersona with all resources loaded and configured.
        """
        config = self._registry.get(persona, context)
        
        # Resolve model
        model = await self._models.get(config.model_id)
        
        # Resolve adapter (if any)
        if config.adapter_name:
            await self._loras.apply_adapter(config.adapter_name)
        
        # Resolve prompt
        system_prompt = await self._prompts.get(
            config.system_prompt_path,
            variables=config.prompt_variables
        )
        
        return ResolvedPersona(
            config=config,
            model=model,
            system_prompt=system_prompt,
            lora_pool=self._loras,
        )


@dataclass
class ResolvedPersona:
    """A fully resolved persona ready for generation."""
    
    config: PersonaConfig
    model: Any  # Model instance
    system_prompt: str
    lora_pool: MultiLoRAPool
    
    async def generate(self, prompt: str) -> str:
        """Generate with this persona's full configuration."""
        # Ensure correct adapter is active
        if self.config.adapter_name:
            await self.lora_pool.apply_adapter(self.config.adapter_name)
        
        # Generate with configured params
        return await self.lora_pool.generate(
            prompt=f"{self.system_prompt}\n\n{prompt}",
            adapter=self.config.adapter_name,
            config=GenerationConfig(
                max_tokens=self.config.max_tokens,
                temperature=self.config.temperature,
            ),
            actor=self.config.name,
        )
```

**Integration with Engine**

```python
class LunaEngine:
    """Main engine with unified persona routing."""
    
    def __init__(self, config: EngineConfig):
        # Resource pools
        self._model_pool = ModelPool(config.models)
        self._lora_pool = MultiLoRAPool()
        self._prompt_cache = PromptCache(config.prompts_dir)
        
        # Persona system
        self._registry = PersonaRegistry(config.personas_dir)
        self._router = PersonaRouter(
            registry=self._registry,
            model_pool=self._model_pool,
            lora_pool=self._lora_pool,
            prompt_cache=self._prompt_cache,
        )
        
        # Context detection
        self._context_detector = ContextDetector()
        self._current_context: Optional[str] = None
    
    async def process_message(self, message: str) -> str:
        """Process a message through the full pipeline."""
        
        # 1. Detect context
        detected = await self._context_detector.detect(message)
        if detected.confidence > 0.7:
            self._current_context = detected.state
        
        # 2. Resolve Luna persona for current context
        luna = await self._router.resolve("luna", context=self._current_context)
        
        # 3. Fetch relevant memories (via Librarian)
        librarian = await self._router.resolve("librarian")
        context = await librarian.generate(f"Find context for: {message}")
        
        # 4. Generate response
        response = await luna.generate(f"Context:\n{context}\n\nUser: {message}")
        
        # 5. Extract memories (via Scribe)
        scribe = await self._router.resolve("scribe")
        extraction = await scribe.generate(f"Extract from:\n{message}\n{response}")
        
        # 6. Store extracted memories
        await self._store_extraction(extraction)
        
        return response
```
