import pandas as pd
from datetime import datetime
import os
from email.mime.text import MIMEText
import smtplib

class VenueInquiryTracker:
    def __init__(self, spreadsheet_path):
        self.spreadsheet_path = spreadsheet_path
        # Create spreadsheet if it doesn't exist
        if not os.path.exists(spreadsheet_path):
            self.create_new_spreadsheet()
        self.df = pd.read_excel(spreadsheet_path)
        
    def create_new_spreadsheet(self):
        """Create a new spreadsheet with proper columns"""
        columns = [
            'Submission Date',
            'Name',
            'Email',
            'Phone',
            'Event Date',
            'Event Type',
            'Guest Count',
            'Budget Range',
            'Special Requests',
            'Status',
            'Follow Up Date',
            'Notes'
        ]
        pd.DataFrame(columns=columns).to_excel(self.spreadsheet_path, index=False)
    
    def add_inquiry(self, inquiry_data):
        """Add new inquiry to spreadsheet"""
        inquiry_data['Submission Date'] = datetime.now()
        inquiry_data['Status'] = 'New'
        
        # Add new row to dataframe
        self.df = pd.concat([self.df, pd.DataFrame([inquiry_data])], ignore_index=True)
        self.save_spreadsheet()
        self.send_confirmation_email(inquiry_data)
        
    def save_spreadsheet(self):
        """Save changes to spreadsheet"""
        self.df.to_excel(self.spreadsheet_path, index=False)
        
    def send_confirmation_email(self, inquiry_data):
        """Send confirmation email to inquirer"""
        # Email settings (replace with your details)
        smtp_server = "smtp.gmail.com"
        smtp_port = 587
        sender_email = "your-email@venue.com"
        password = "your-app-specific-password"
        
        # Create email message
        msg = MIMEText(f"""
        Dear {inquiry_data['Name']},
        
        Thank you for your inquiry about our venue. We have received your request for an event on {inquiry_data['Event Date']}.
        
        Our team will review your inquiry and get back to you within 24 hours.
        
        Best regards,
        Your Venue Team
        """)
        
        msg['Subject'] = 'Venue Inquiry Received'
        msg['From'] = sender_email
        msg['To'] = inquiry_data['Email']
        
        # Send email
        try:
            with smtplib.SMTP(smtp_server, smtp_port) as server:
                server.starttls()
                server.login(sender_email, password)
                server.send_message(msg)
        except Exception as e:
            print(f"Failed to send confirmation email: {e}")
            
    def get_pending_inquiries(self):
        """Get all pending inquiries"""
        return self.df[self.df['Status'] == 'New']
    
    def update_status(self, email, new_status, notes=''):
        """Update inquiry status and add notes"""
        idx = self.df[self.df['Email'] == email].index
        if len(idx) > 0:
            self.df.loc[idx[0], 'Status'] = new_status
            if notes:
                self.df.loc[idx[0], 'Notes'] = notes
            self.save_spreadsheet()
