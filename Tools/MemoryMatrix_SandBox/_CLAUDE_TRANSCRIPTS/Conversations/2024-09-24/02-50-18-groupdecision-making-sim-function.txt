    tolerance = tolerance_factors.get(member['alcohol_tolerance'], 1.0)
    
    return base_factor * tolerance

def identify_decision_maker(group_members):
    """Identify the decision maker based on wealth, influence, and sobriety"""
    if not all(member['age'] >= 21 for member in group_members):
        raise ValueError("All group members must be 21+")
    
    potential_dms = []
    for member in group_members:
        # Consider wealth, social influence, and inverse of alcohol level
        sobriety_factor = 1 / calculate_bac_influence(member)  # Less drunk = more qualified to decide
        influence_score = (
            member['wealth_score'] * 0.5 + 
            member['social_influence'] * 0.3 +
            sobriety_factor * 0.2  # Sobriety now factors into decision maker selection
        )
        potential_dms.append((influence_score, member))
    
    return max(potential_dms, key=lambda x: x[0])[1]

def assess_group_mood(group_members):
    """Calculate collective mood of the group with alcohol influence"""
    mood_weights = {
        'excited': 1.2,
        'happy': 1.0,
        'neutral': 0.8,
        'tired': 0.6,
        'irritated': 0.4
    }
    
    total_weight = 0
    weighted_mood = 0
    
    for member in group_members:
        base_weight = mood_weights.get(member['mood'], 0.8)
        alcohol_factor = calculate_bac_influence(member)
        # Alcohol tends to amplify existing moods
        mood_score = base_weight * alcohol_factor
        weighted_mood += mood_score
        total_weight += 1
        
    return weighted_mood / total_weight

def evaluate_personal_mood(decision_maker):
    """Evaluate decision maker's mood including alcohol effects"""
    base_mood = decision_maker['mood_score']  # 0-1 scale
    
    time_multipliers = {
        'morning': 0.9,
        'afternoon': 1.0,
        'evening': 1.1,
        'late_night': 0.8
    }
    
    energy_multipliers = {
        'energetic': 1.2,
        'normal': 1.0,
        'tired': 0.8,
        'exhausted': 0.6
    }
    
    alcohol_factor = calculate_bac_influence(decision_maker)
    
    return (base_mood * 
            time_multipliers.get(decision_maker['time_of_day'], 1.0) *
            energy_multipliers.get(decision_maker['energy_level'], 1.0) *
            alcohol_factor)

def determine_personality_type(decision_maker):
    """Map personality traits and drinking state to personality type"""
    personality_scores = {
        "Big Spender": 0,
        "Frugal Fred": 0,
        "Impulse Buyer": 0,
        "Overthink Vince": 0,
        "Trendsetter Tina": 0,
        "Eco-Warrior Ethan": 0,
        "Bargain Hunter Betty": 0,
        "Luxury Lover Lucy": 0,
        "Practical Pete": 0,
        "Social Shopper Sam": 0
    }
    
    traits = decision_maker['personality_traits']
    alcohol_factor = calculate_bac_influence(decision_maker)
    
    # Alcohol affects personality type weights
    for trait, value in traits.items():
        if trait == 'spending_frequency':
            # More drinks = more likely to be a big spender
            personality_scores["Big Spender"] += value * 2 * alcohol_factor
            personality_scores["Frugal Fred"] += (1 - value) * 2 / alcohol_factor
        elif trait == 'impulsiveness':
            # More drinks = more impulsive
            personality_scores["Impulse Buyer"] += value * 2 * alcohol_factor
            personality_scores["Overthink Vince"] += (1 - value) * 2 / alcohol_factor
            
    return max(personality_scores.items(), key=lambda x: x[1])[0]

def calculate_purchase_likelihood(decision_maker, group_members, item):
    """Calculate purchase likelihood with alcohol factors"""
    personality = determine_personality_type(decision_maker)
    group_mood = assess_group_mood(group_members)
    personal_mood = evaluate_personal_mood(decision_maker)
    
    # Personality thresholds now affected by alcohol
    base_personality_thresholds = {
        "Big Spender": 0.6,
        "Frugal Fred": 0.9,
        "Impulse Buyer": 0.5,
        "Overthink Vince": 0.8,
        "Trendsetter Tina": 0.65,
        "Eco-Warrior Ethan": 0.7,
        "Bargain Hunter Betty": 0.75,
        "Luxury Lover Lucy": 0.7,
        "Practical Pete": 0.8,
        "Social Shopper Sam": 0.7
    }
    
    # Alcohol lowers all thresholds
    alcohol_factor = calculate_bac_influence(decision_maker)
    base_threshold = base_personality_thresholds.get(personality, 0.7)
    adjusted_threshold = base_threshold / alcohol_factor  # Lower threshold when drinking
    
    # Calculate purchase score
    purchase_score = (
        personal_mood * 0.4 +
        group_mood * 0.3 +
        (1 - (item['price'] / decision_maker['max_spend'])) * 0.3
    )
    
    return {
        'will_purchase': purchase_score > adjusted_threshold,
        'purchase_score': purchase_score,
        'threshold': adjusted_threshold,
        'dominant_personality': personality,
        'group_mood_score': group_mood,
        'personal_mood_score': personal_mood,
        'alcohol_influence': alcohol_factor
    }
