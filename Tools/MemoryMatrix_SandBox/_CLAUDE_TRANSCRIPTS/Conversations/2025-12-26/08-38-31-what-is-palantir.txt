## The Sequencer

Orchestrates the pipeline:

```python
class ConversationSequencer:
    def __init__(self):
        self.bus = EventBus()
        self.state = ConversationState.IDLE
        self.pending_tasks: dict[str, Task] = {}
        
    async def run(self):
        async for event in self.bus.stream():
            await self.handle(event)
    
    async def handle(self, event: Event):
        match (self.state, event.type):
            
            # User starts talking
            case (ConversationState.IDLE, EventType.AUDIO_CHUNK):
                self.state = ConversationState.LISTENING
                self.start_task("transcribe", self.transcribe(event.data))
                
            # Got partial transcript - start speculating
            case (ConversationState.LISTENING, EventType.TRANSCRIPT_PARTIAL):
                self.start_task("speculate", self.speculate(event.data))
                
            # Final transcript - go
            case (ConversationState.LISTENING, EventType.TRANSCRIPT_FINAL):
                self.cancel_task("speculate")
                self.state = ConversationState.PROCESSING
                await self.process_input(event.data)
                
            # User interrupts while Luna speaking
            case (ConversationState.SPEAKING, EventType.USER_INTERRUPT):
                await self.yield_floor()
                self.state = ConversationState.LISTENING
                
            # Ready to speak
            case (ConversationState.PROCESSING, EventType.TOKENS_READY):
                self.state = ConversationState.SPEAKING
                self.start_task("speak", self.speak(event.data))
                
            # etc...
    
    async def process_input(self, transcript: str):
        # Fire parallel operations
        intent_task = self.start_task("intent", self.classify(transcript))
        memory_task = self.start_task("memory", self.search_memory(transcript))
        
        # Wait for both
        intent, memories = await asyncio.gather(
            intent_task, memory_task
        )
        
        # Quick response if we have a template
        if intent.confidence > 0.9 and intent.type in self.cache.templates:
            # Instant response
            await self.bus.emit(Event(
                type=EventType.TOKENS_READY,
                data=self.cache.get(intent.type)
            ))
        else:
            # Full generation
            self.start_task("generate", self.generate(transcript, intent, memories))
```
