```javascript
function createFolderStructure() {
  var root = DriveApp.getRootFolder();
  var mainFolder = root.createFolder("Dynamic Display Creative");
  
  // Create main categories
  var projects = mainFolder.createFolder("Projects");
  var services = mainFolder.createFolder("Services");
  var administration = mainFolder.createFolder("Administration");
  var marketing = mainFolder.createFolder("Marketing");
  
  // Projects subfolders
  var projectTypes = ["Special Events", "Corporate Events", "Festivals", "Concerts"];
  projectTypes.forEach(function(type) {
    var typeFolder = projects.createFolder(type);
    typeFolder.createFolder("Project Files");
    typeFolder.createFolder("Client Assets");
    typeFolder.createFolder("Designs");
  });
  
  // Services subfolders
  var serviceTypes = {
    "Construction": ["Equipment Inventory", "Safety Protocols", "Construction Plans"],
    "Consultation": [],
    "Rental": ["Rental Inventory", "Rental Agreements", "Maintenance Schedules"],
    "Design": ["Design Templates", "3D Models", "Material Library"]
  };
  
  for (var service in serviceTypes) {
    var serviceFolder = services.createFolder(service);
    serviceTypes[service].forEach(function(subFolder) {
      serviceFolder.createFolder(subFolder);
    });
  }
  
  // Administration subfolders
  ["Human Resources", "Finance", "Legal"].forEach(function(folder) {
    administration.createFolder(folder);
  });
  
  // Marketing subfolders
  ["Portfolio", "Social Media", "Press Releases"].forEach(function(folder) {
    marketing.createFolder(folder);
  });
  
  Logger.log("Folder structure created successfully!");
}
```
