## Entity System Architecture

### Core Concept

**Entities** are first-class objects in Luna Engine — people, personas, places, projects. They have:
- Canonical profiles (curated, authoritative)
- Accumulated knowledge (from Memory Matrix)
- Relationships (graph edges to other entities)
- Context rules (when/how they load into Luna's working memory)

---

### Schema

```sql
-- Core entity storage
CREATE TABLE entities (
    id TEXT PRIMARY KEY,           -- slug: 'ahab', 'marzipan', 'ben-franklin'
    entity_type TEXT NOT NULL,     -- 'person' | 'persona' | 'place' | 'project'
    name TEXT NOT NULL,
    aliases TEXT,                  -- JSON: ["Zayne", "Ahab"]
    
    -- Structured profile (always available)
    core_facts TEXT,               -- JSON: compact, ~500 tokens max
    
    -- Extended profile (loaded on demand)
    full_profile TEXT,             -- Markdown, can be lengthy
    
    -- For personas: voice/behavior parameters
    voice_config TEXT,             -- JSON: tone, patterns, constraints
    
    metadata TEXT,                 -- Flexible JSON blob
    created_at TEXT DEFAULT CURRENT_TIMESTAMP,
    updated_at TEXT DEFAULT CURRENT_TIMESTAMP
);

-- Relationship graph
CREATE TABLE entity_relationships (
    id INTEGER PRIMARY KEY,
    from_entity TEXT REFERENCES entities(id),
    to_entity TEXT REFERENCES entities(id),
    relationship TEXT NOT NULL,    -- 'creator', 'friend', 'collaborator', 'embodies'
    strength REAL DEFAULT 0.5,     -- 0-1, for relevance weighting
    bidirectional INTEGER DEFAULT 0,
    context TEXT,                  -- "Met at Mars College 2025"
    created_at TEXT DEFAULT CURRENT_TIMESTAMP
);

-- Link entities to memory nodes (for accumulated knowledge)
CREATE TABLE entity_mentions (
    entity_id TEXT REFERENCES entities(id),
    node_id TEXT REFERENCES memory_nodes(id),
    mention_type TEXT,             -- 'subject', 'author', 'reference'
    confidence REAL DEFAULT 1.0,
    PRIMARY KEY (entity_id, node_id)
);

-- Indexes
CREATE INDEX idx_entities_type ON entities(entity_type);
CREATE INDEX idx_relationships_from ON entity_relationships(from_entity);
CREATE INDEX idx_relationships_to ON entity_relationships(to_entity);
CREATE INDEX idx_mentions_entity ON entity_mentions(entity_id);
```

## Entity Versioning System

### Design Goals

1. **Track how understanding evolves** — "Marzipan moved from solar to AI focus"
2. **Never lose history** — Scribe can update, but old versions persist
3. **Enable temporal queries** — "What did Luna know about X in December?"
4. **Support rollback** — If Scribe makes a bad update, Librarian can revert

### Schema Addition

```sql
-- ============================================================
-- ENTITY VERSIONS: Full history of profile changes
-- ============================================================
CREATE TABLE entity_versions (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    entity_id TEXT NOT NULL REFERENCES entities(id),
    version INTEGER NOT NULL,                -- 1, 2, 3...
    
    -- Snapshot of entity state at this version
    core_facts TEXT,                         -- JSON snapshot
    full_profile TEXT,                       -- Markdown snapshot
    voice_config TEXT,                       -- JSON snapshot (for personas)
    
    -- Change metadata
    change_type TEXT NOT NULL,               -- 'create' | 'update' | 'synthesize' | 'rollback'
    change_summary TEXT,                     -- "Added Mars College location"
    changed_by TEXT,                         -- 'scribe' | 'librarian' | 'manual'
    change_source TEXT,                      -- node_id or conversation_id that triggered
    
    -- Timestamps
    created_at TEXT DEFAULT CURRENT_TIMESTAMP,
    valid_from TEXT DEFAULT CURRENT_TIMESTAMP,
    valid_until TEXT,                        -- NULL = current version
    
    UNIQUE(entity_id, version)
);

-- Fast lookup of current version
CREATE INDEX idx_entity_versions_current 
ON entity_versions(entity_id, valid_until) 
WHERE valid_until IS NULL;

-- Temporal queries
CREATE INDEX idx_entity_versions_temporal 
ON entity_versions(entity_id, valid_from, valid_until);
```

### Scribe's Update Protocol

```python
async def update_entity(
    entity_id: str,
    changes: dict,
    change_summary: str,
    source_id: Optional[str] = None
) -> EntityVersion:
    """
    Scribe updates an entity, creating a new version.
    
    NEVER overwrites — always appends to version history.
    """
    # 1. Get current version
    current = await get_current_version(entity_id)
    
    # 2. Close current version
    await db.execute("""
        UPDATE entity_versions 
        SET valid_until = CURRENT_TIMESTAMP
        WHERE entity_id = ? AND valid_until IS NULL
    """, entity_id)
    
    # 3. Merge changes into snapshot
    new_core_facts = merge_json(current.core_facts, changes.get("core_facts", {}))
    new_full_profile = changes.get("full_profile") or current.full_profile
    
    # 4. Create new version
    new_version = EntityVersion(
        entity_id=entity_id,
        version=current.version + 1,
        core_facts=new_core_facts,
        full_profile=new_full_profile,
        voice_config=current.voice_config,  # Usually unchanged
        change_type="update",
        change_summary=change_summary,
        changed_by="scribe",
        change_source=source_id,
    )
    
    await db.insert("entity_versions", new_version)
    
    # 5. Update entities table (current view)
    await db.execute("""
        UPDATE entities 
        SET core_facts = ?, full_profile = ?, updated_at = CURRENT_TIMESTAMP
        WHERE id = ?
    """, new_core_facts, new_full_profile, entity_id)
    
    return new_version
```

### Temporal Queries

```python
async def get_entity_at_time(entity_id: str, timestamp: datetime) -> Optional[EntityVersion]:
    """What did Luna know about this entity at a specific time?"""
    return await db.query_one("""
        SELECT * FROM entity_versions
        WHERE entity_id = ?
          AND valid_from <= ?
          AND (valid_until IS NULL OR valid_until > ?)
    """, entity_id, timestamp, timestamp)

async def get_entity_history(entity_id: str) -> List[EntityVersion]:
    """Get full version history of an entity."""
    return await db.query("""
        SELECT * FROM entity_versions
        WHERE entity_id = ?
        ORDER BY version ASC
    """, entity_id)

async def get_recent_changes(days: int = 7) -> List[EntityVersion]:
    """What entities changed recently?"""
    return await db.query("""
        SELECT * FROM entity_versions
        WHERE created_at > datetime('now', ?)
        ORDER BY created_at DESC
    """, f"-{days} days")
```

### Librarian's Rollback

```python
async def rollback_entity(entity_id: str, to_version: int, reason: str) -> EntityVersion:
    """
    Librarian can rollback to a previous version.
    
    Doesn't delete history — creates a NEW version that copies the old state.
    """
    # 1. Get the version to restore
    old_version = await db.query_one("""
        SELECT * FROM entity_versions
        WHERE entity_id = ? AND version = ?
    """, entity_id, to_version)
    
    if not old_version:
        raise ValueError(f"Version {to_version} not found for {entity_id}")
    
    # 2. Close current version
    await db.execute("""
        UPDATE entity_versions 
        SET valid_until = CURRENT_TIMESTAMP
        WHERE entity_id = ? AND valid_until IS NULL
    """, entity_id)
    
    # 3. Get latest version number
    latest = await db.query_one("""
        SELECT MAX(version) as v FROM entity_versions WHERE entity_id = ?
    """, entity_id)
    
    # 4. Create rollback version (copy of old state, new version number)
    rollback_version = EntityVersion(
        entity_id=entity_id,
        version=latest.v + 1,
        core_facts=old_version.core_facts,
        full_profile=old_version.full_profile,
        voice_config=old_version.voice_config,
        change_type="rollback",
        change_summary=f"Rollback to v{to_version}: {reason}",
        changed_by="librarian",
        change_source=None,
    )
    
    await db.insert("entity_versions", rollback_version)
    
    # 5. Update entities table
    await db.execute("""
        UPDATE entities 
        SET core_facts = ?, full_profile = ?, voice_config = ?, updated_at = CURRENT_TIMESTAMP
        WHERE id = ?
    """, old_version.core_facts, old_version.full_profile, old_version.voice_config, entity_id)
    
    return rollback_version
```

    '{"change_type": "update"}'

-- Entity versions can also supersede each other
INSERT INTO edges (source, target, edge_type)
VALUES ('marzipan_v2', 'marzipan_v1', 'supersedes');
