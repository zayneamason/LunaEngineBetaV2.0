    def emit(cls, event_type, node, old_val, new_val):
        for listener in cls.listeners:
            listener(event_type, node, old_val, new_val)
        # Also log to file for now
        log.info(f"EVENT: {event_type} {node.id} {old_val} → {new_val}")
