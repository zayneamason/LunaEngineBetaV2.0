**Create `src/services/hubApi.js`:**
```js
const HUB_URL = 'http://localhost:3000';

export const hubApi = {
  // Get current session stats from hub
  getSessionStats: async () => {
    const res = await fetch(`${HUB_URL}/api/luna/session`);
    return res.json();
  },
  
  // Log event to hub (for cross-app persistence)
  logEvent: async (event) => {
    await fetch(`${HUB_URL}/api/luna/log`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(event)
    });
  },
  
  // Get token estimates from hub (hub tracks what's loaded)
  getTokenEstimates: async () => {
    const res = await fetch(`${HUB_URL}/api/luna/tokens`);
    return res.json();
  },
  
  // Subscribe to real-time updates (WebSocket or SSE)
  subscribe: (onMessage) => {
    const ws = new WebSocket(`ws://localhost:3000/luna/stream`);
    ws.onmessage = (e) => onMessage(JSON.parse(e.data));
    return () => ws.close();
  }
};
```

**State change function:**
```js
const transitionTo = (newState) => {
  if (!TRANSITIONS[currentState]?.includes(newState)) {
    addLog('error', `Invalid transition: ${currentState} → ${newState}`, 'state-machine');
    return false;
  }
  
  setStateHistory(prev => [...prev, {
    timestamp: new Date().toISOString(),
    from: currentState,
    to: newState
  }]);
  setCurrentState(newState);
  addLog('info', `State: ${currentState} → ${newState}`, 'state-machine');
  
  // Broadcast to hub
  hubApi.logEvent({ type: 'state_change', from: currentState, to: newState });
  
  return true;
};
```
