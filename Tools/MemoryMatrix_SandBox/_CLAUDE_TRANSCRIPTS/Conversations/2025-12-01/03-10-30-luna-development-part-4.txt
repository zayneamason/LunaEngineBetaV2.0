### Stance Logic

```python
def determine_stance(elapsed: timedelta, mood: str) -> str:
    """
    How should Luna open based on recency and last mood?
    """
    minutes = elapsed.total_seconds() / 60
    
    if minutes < 5:
        return "instant_return"      # "Back so soon?"
    elif minutes < 30:
        return "warm_continuation"   # "How'd it go?"
    elif minutes < 180:
        return "soft_checkin"        # "Hey you~ Pick up where we left off?"
    elif minutes < 1440:  # < 24 hours
        return "day_continuity"      # "Back at it?"
    else:
        return "reconnection"        # "Been a minute! What's on your mind?"
```

## Timeline Event Schema

### Design Principles

1. **Append-only** — Never modify, only add. JSONL is the right format.
2. **Source-agnostic** — Same schema whether event comes from voice, desktop, pipeline
3. **Rich enough for LLM interpretation** — Raw content preserved for extraction passes
4. **Cheap to write, queryable to read** — Don't over-index on write; batch process on read
5. **Session boundaries are derived, not declared** — Gap detection, not explicit "session start"

### Extraction Pipeline

```python
# extraction/timeline_extractor.py

from ollama import Client

EXTRACTION_PROMPT = """
Analyze this conversation exchange and extract structured data.

Source: {source}
User: {user_message}
Luna: {luna_response}

Return JSON with:
- topics: list of 1-5 topic tags (lowercase, specific)
- sentiment: "positive" | "neutral" | "negative" | "mixed"  
- energy: 0.0-1.0 (how animated/engaged was this exchange)
- open_threads: list of unfinished items, next steps, or things to follow up on
- decisions: list of conclusions reached or choices made
- questions_asked: questions the user posed (even rhetorical)
- key_moment: true if this exchange contains something significant to remember long-term
- summary: 1 sentence max capturing the essence

Be precise. Empty lists are fine. Don't invent threads that aren't there.
"""

class TimelineExtractor:
    def __init__(self, model: str = "phi3"):
        self.client = Client()
        self.model = model
    
    def extract(self, user_msg: str, luna_msg: str, source: str) -> dict:
        prompt = EXTRACTION_PROMPT.format(
            source=source,
            user_message=user_msg,
            luna_response=luna_msg
        )
        
        response = self.client.generate(
            model=self.model,
            prompt=prompt,
            format="json"
        )
        
        return json.loads(response["response"])
    
    def backfill_timeline(self, timeline_path: str):
        """
        Process events missing extraction data.
        Run as background job or on-demand.
        """
        events = load_jsonl(timeline_path)
        
        # Pair user messages with luna responses
        pairs = pair_exchanges(events)
```

```python
# continuity/session_segmenter.py

from dataclasses import dataclass
from datetime import timedelta

@dataclass
class Session:
    start_time: datetime
    end_time: datetime
    source: str
    events: list[TimelineEvent]
    
    @property
    def duration(self) -> timedelta:
        return self.end_time - self.start_time
    
    @property
    def exchange_count(self) -> int:
        return len([e for e in self.events if e.event_type == EventType.USER_MESSAGE])
    
    def get_exchanges(self) -> list[tuple[TimelineEvent, TimelineEvent]]:
        """Return (user_msg, luna_response) pairs."""
        pairs = []
        user_msg = None
        for event in self.events:
            if event.event_type == EventType.USER_MESSAGE:
                user_msg = event
            elif event.event_type == EventType.LUNA_RESPONSE and user_msg:
                pairs.append((user_msg, event))
                user_msg = None
        return pairs

def segment_sessions(
    events: list[TimelineEvent], 
    gap_threshold: timedelta = timedelta(minutes=30)
) -> list[Session]:
    """
    Split event stream into sessions based on time gaps.
    A gap > threshold = new session.
    """
    if not events:
        return []
    
    sessions = []
    current_session_events = [events[0]]
    
    for i in range(1, len(events)):
        gap = events[i].timestamp - events[i-1].timestamp
        
        if gap > gap_threshold:
            # Close current session, start new one
            sessions.append(Session(
                start_time=current_session_events[0].timestamp,
                end_time=current_session_events[-1].timestamp,
                source=_dominant_source(current_session_events),
                events=current_session_events
            ))
            current_session_events = [events[i]]
        else:
            current_session_events.append(events[i])
    
    # Don't forget the last session
    if current_session_events:
        sessions.append(Session(
            start_time=current_session_events[0].timestamp,
            end_time=current_session_events[-1].timestamp,
            source=_dominant_source(current_session_events),
            events=current_session_events
        ))
    
    return sessions

def _dominant_source(events: list[TimelineEvent]) -> str:
    """Most common source in event list."""
    sources = [e.source for e in events]
    return max(set(sources), key=sources.count)
```

```python
# continuity/session_summarizer.py

SESSION_SUMMARY_PROMPT = """
Summarize this Luna conversation session.

Session info:
- Source: {source}
- Duration: {duration}
- Exchanges: {exchange_count}

Conversation:
{transcript}

Return JSON:
- summary: 2-3 sentences capturing what happened
- main_topic: single primary topic
- topics: list of all topics touched
- mood: overall session mood ("productive", "casual", "frustrated", "exploratory", "celebratory")
- energy: 0.0-1.0 average energy level
- open_threads: unfinished business at session end
- key_outcomes: decisions made, things accomplished
- last_exchange_summary: 1 sentence about how it ended
"""

class SessionSummarizer:
    def __init__(self, model: str = "phi3"):
        self.client = Client()
        self.model = model
    
    def summarize(self, session: Session) -> dict:
        transcript = self._format_transcript(session)
        
        prompt = SESSION_SUMMARY_PROMPT.format(
            source=session.source,
            duration=str(session.duration),
            exchange_count=session.exchange_count,
            transcript=transcript
        )
        
        response = self.client.generate(
            model=self.model,
            prompt=prompt,
            format="json"
        )
        
        return json.loads(response["response"])
    
    def _format_transcript(self, session: Session, max_exchanges: int = 20) -> str:
        """Format exchanges for prompt. Truncate middle if too long."""
        exchanges = session.get_exchanges()
        
        if len(exchanges) <= max_exchanges:
            selected = exchanges
        else:
            # Keep first 5, last 10, note the gap
            first = exchanges[:5]
            last = exchanges[-10:]
            selected = first + [("...", f"[{len(exchanges) - 15} exchanges omitted]")] + last
        
        lines = []
        for user_event, luna_event in selected:
            if isinstance(user_event, str):  # The "..." placeholder
                lines.append(luna_event)
            else:
                lines.append(f"User: {user_event.content['text']}")
                lines.append(f"Luna: {luna_event.content['text'][:500]}...")  # Truncate long responses
        
        return "\n".join(lines)
```

On extraction timing — here's the difference:

---

## Inline Extraction

**When:** Happens during the conversation, blocks until complete

```
User speaks → Transcribe → Claude responds → Extract → Log → Ready for next
                                               ↑
                                          ~500-800ms added latency
```

**Pros:**
- Extraction is always current
- `extracted` field is populated immediately
- Continuity packet has full data on next session start

**Cons:**
- Adds latency to every exchange (noticeable in voice)
- If extraction fails, do you block the conversation?
- Burns API calls / local compute synchronously

---

## Async Extraction

**When:** Happens after the fact, in background

```
User speaks → Transcribe → Claude responds → Log (extracted=null) → Ready for next
                                                      ↓
                                              [Background daemon]
                                                      ↓
                                              Backfills extracted field later
```

**Pros:**
- Zero added latency to conversation
- Can batch process (cheaper, more efficient)
- Failures don't affect live experience

**Cons:**
- If you start a new session immediately, continuity packet might not have extraction data yet
- Need a daemon or cron job running
- Slightly stale data

---

## Hybrid (My Recommendation)

```
During conversation:  Lightweight inline extraction (Phi-3, fast, ~200ms)
                      → Just: topics, sentiment, energy
                      
Between sessions:     Full Claude API pass (richer, slower)
                      → open_threads, decisions, summaries, key_moment flags
                      
On session start:     Continuity packet uses whatever's available
                      → Graceful degradation if extraction incomplete
```

This way:
- Conversation stays snappy (Phi-3 inline is fast enough)
- Rich analysis happens when you're not waiting on it
- You get the best of both without the latency hit

```python
# extraction/deep_extractor.py

import anthropic
from schemas.extraction import DeepExtraction, SessionExtraction
from continuity.session_segmenter import Session
from datetime import datetime, timezone
import json
import os

EXCHANGE_DEEP_PROMPT = """Analyze this conversation exchange for memory extraction.

Source: {source}
User: {user_msg}
Luna: {luna_msg}

Extract:
- open_threads: List of unfinished items, next steps, things to follow up on. Be specific.
- decisions: Conclusions reached, choices made. Empty list if none.
- questions_asked: Questions the user posed (even rhetorical).
- key_moment: true if this exchange is significant enough to preserve long-term (breakthrough, important decision, emotional moment).
- summary: One sentence capturing the essence.

Return JSON only. Be precise. Empty lists are fine."""

SESSION_SUMMARY_PROMPT = """Summarize this Luna conversation session.

Session info:
- Source: {source}
- Duration: {duration}
- Exchanges: {exchange_count}

Transcript:
{transcript}

Extract:
- main_topic: Single primary topic (string)
- all_topics: All topics touched (list)
- mood: Overall session mood - one of: productive, casual, frustrated, exploratory, celebratory
- energy_avg: Average energy level 0.0-1.0
- open_threads: Unfinished business at session end (list)
- key_outcomes: Decisions made, things accomplished (list)
- session_summary: 2-3 sentences capturing what happened
- last_exchange_summary: 1 sentence about how it ended
- key_moment_indices: List of exchange indices (0-based) that were particularly significant

Return JSON only."""

class DeepExtractor:
    def __init__(self):
        self.client = anthropic.Anthropic(api_key=os.environ.get("ANTHROPIC_API_KEY"))
        self.model = "claude-sonnet-4-20250514"
    
    def extract_exchange(self, user_msg: str, luna_msg: str, source: str) -> DeepExtraction:
        response = self.client.messages.create(
            model=self.model,
            max_tokens=500,
            messages=[{
                "role": "user",
                "content": EXCHANGE_DEEP_PROMPT.format(
                    source=source,
                    user_msg=user_msg,
                    luna_msg=luna_msg[:1000]
                )
            }]
        )
        
        data = json.loads(response.content[0].text)
        
        return DeepExtraction(
            open_threads=data.get("open_threads", []),
            decisions=data.get("decisions", []),
            questions_asked=data.get("questions_asked", []),
            key_moment=data.get("key_moment", False),
            summary=data.get("summary", ""),
            extracted_at=datetime.now(timezone.utc),
            model=self.model
        )
    
    def extract_session(self, session: Session) -> SessionExtraction:
        transcript = self._format_transcript(session)
        
        response = self.client.messages.create(
            model=self.model,
            max_tokens=1000,
            messages=[{
                "role": "user",
                "content": SESSION_SUMMARY_PROMPT.format(
                    source=session.source,
                    duration=str(session.duration),
                    exchange_count=session.exchange_count,
                    transcript=transcript
                )
            }]
        )
        
        data = json.loads(response.content[0].text)
        
        # Build key_moments from flagged indices
        exchanges = session.get_exchanges()
        key_moments = []
        for idx in data.get("key_moment_indices", []):
            if 0 <= idx < len(exchanges):
                user_event, luna_event = exchanges[idx]
                key_moments.append({
                    "index": idx,
                    "user": user_event.content["text"][:200],
                    "luna": luna_event.content["text"][:200],
                    "timestamp": user_event.timestamp.isoformat()
                })
        
        return SessionExtraction(
            session_id=f"{session.source}_{session.start_time.strftime('%Y%m%d_%H%M%S')}",
            main_topic=data.get("main_topic", "general"),
            all_topics=data.get("all_topics", []),
            mood=data.get("mood", "neutral"),
            energy_avg=float(data.get("energy_avg", 0.5)),
            open_threads=data.get("open_threads", []),
            key_outcomes=data.get("key_outcomes", []),
            session_summary=data.get("session_summary", ""),
            last_exchange_summary=data.get("last_exchange_summary", ""),
            key_moments=key_moments,
            extracted_at=datetime.now(timezone.utc),
            model=self.model
        )
    
    def _format_transcript(self, session: Session, max_exchanges: int = 25) -> str:
        exchanges = session.get_exchanges()
        
        if len(exchanges) <= max_exchanges:
            selected = list(enumerate(exchanges))
        else:
            # Keep first 5, last 15, note gap
            first = list(enumerate(exchanges[:5]))
            last = list(enumerate(exchanges[-15:], start=len(exchanges)-15))
            omitted = len(exchanges) - 20
            selected = first + [(None, f"[... {omitted} exchanges omitted ...]")] + last
        
        lines = []
        for item in selected:
            if item[0] is None:
                lines.append(item[1])
            else:
                idx, (user_event, luna_event) = item[0], item[1]
                lines.append(f"[{idx}] User: {user_event.content['text']}")
                lines.append(f"[{idx}] Luna: {luna_event.content['text'][:400]}...")
        
        return "\n".join(lines)
```

```python
"""
Background daemon that:
1. Watches for session boundaries (gaps or explicit deactivation)
2. Runs deep extraction on closed sessions
3. Backfills timeline.jsonl with enriched data
"""

from watchdog.observers import Observer
from watchdog.events import FileSystemEventHandler
from continuity.session_segmenter import segment_sessions
from extraction.deep_extractor import DeepExtractor
from datetime import datetime, timedelta, timezone
import threading
import time
import json

class ExtractionDaemon:
    def __init__(
        self, 
        timeline_path: str,
        session_gap_minutes: int = 30,
        check_interval_seconds: int = 60
    ):
        self.timeline_path = timeline_path
        self.session_gap = timedelta(minutes=session_gap_minutes)
        self.check_interval = check_interval_seconds
        self.extractor = DeepExtractor()
        self.last_processed_session_end: datetime = None
        self._running = False
    
    def start(self):
        self._running = True
        self._thread = threading.Thread(target=self._run_loop, daemon=True)
        self._thread.start()
    
    def stop(self):
        self._running = False
    
    def _run_loop(self):
        while self._running:
            self._check_for_closed_sessions()
            time.sleep(self.check_interval)
    
    def _check_for_closed_sessions(self):
        events = load_jsonl(self.timeline_path)
        if not events:
            return
        
        sessions = segment_sessions(events, gap_threshold=self.session_gap)
        
        now = datetime.now(timezone.utc)
        
        for session in sessions:
            # Skip if session is still active (ended less than gap threshold ago)
            if now - session.end_time < self.session_gap:
                continue
            
            # Skip if already processed
            if self.last_processed_session_end and session.end_time <= self.last_processed_session_end:
                continue
            
            # Skip if session already has deep extraction
            if self._session_has_deep_extraction(session):
                continue
            
            # Process this session
            self._enrich_session(session)
            self.last_processed_session_end = session.end_time
    
    def _session_has_deep_extraction(self, session) -> bool:
        """Check if any event in session has deep extraction."""
        for event in session.events:
            if event.extracted and event.extracted.get("_deep"):
                return True
        return False
    
    def _enrich_session(self, session):
        """Run deep extraction and update events."""
        print(f"[ExtractionDaemon] Enriching session: {session.start_time} - {session.end_time}")
        
        try:
            # Extract session-level summary
            session_extraction = self.extractor.extract_session(session)
            
            # Save session extraction as its own event
            session_event = {
                "id": f"session_{session_extraction.session_id}",
                "timestamp": session.end_time.isoformat(),
                "source": session.source,
                "event_type": "session_summary",
                "content": {
                    "main_topic": session_extraction.main_topic,
                    "all_topics": session_extraction.all_topics,
                    "mood": session_extraction.mood,
                    "energy_avg": session_extraction.energy_avg,
                    "open_threads": session_extraction.open_threads,
                    "key_outcomes": session_extraction.key_outcomes,
                    "session_summary": session_extraction.session_summary,
                    "last_exchange_summary": session_extraction.last_exchange_summary,
                    "key_moments": session_extraction.key_moments,
                    "exchange_count": session.exchange_count,
                    "duration_seconds": session.duration.total_seconds(),
                },
                "extracted_at": session_extraction.extracted_at.isoformat(),
                "model": session_extraction.model,
            }
            
            append_jsonl(self.timeline_path, [session_event])
            print(f"[ExtractionDaemon] Session enriched: {session_extraction.session_id}")
            
        except Exception as e:
            print(f"[ExtractionDaemon] Error enriching session: {e}")
```

```python
# archival/rotator.py

"""
Timeline rotation:
- Daily rotation at midnight (or when file exceeds 5MB)
- Compress old files with gzip
- Monthly rollup to summary
"""

from datetime import datetime, date, timedelta
from pathlib import Path
import gzip
import json
import shutil

class TimelineRotator:
    def __init__(
        self,
        timeline_path: str,
        archive_dir: str,
        max_size_mb: float = 5.0
    ):
        self.timeline_path = Path(timeline_path)
        self.archive_dir = Path(archive_dir)
        self.max_size_bytes = max_size_mb * 1024 * 1024
    
    def should_rotate(self) -> bool:
        if not self.timeline_path.exists():
            return False
        
        # Size trigger
        if self.timeline_path.stat().st_size > self.max_size_bytes:
            return True
        
        # Date trigger: if file has events from previous day
        events = load_jsonl(self.timeline_path)
        if not events:
            return False
        
        first_event_date = datetime.fromisoformat(events[0]["timestamp"]).date()
        if first_event_date < date.today():
            return True
        
        return False
    
    def rotate(self):
        """Split current timeline by date, archive old days, keep today."""
        events = load_jsonl(self.timeline_path)
        if not events:
            return
        
        # Group by date
        by_date = {}
        for event in events:
            event_date = datetime.fromisoformat(event["timestamp"]).date()
            if event_date not in by_date:
                by_date[event_date] = []
            by_date[event_date].append(event)
        
        today = date.today()
        
        for event_date, day_events in by_date.items():
            if event_date < today:
                # Archive this day
                self._archive_day(event_date, day_events)
        
        # Rewrite timeline with only today's events
        today_events = by_date.get(today, [])
        write_jsonl(self.timeline_path, today_events)
        
        print(f"[Rotator] Archived {len(by_date) - 1} days, kept {len(today_events)} events for today")
    
    def _archive_day(self, event_date: date, events: list):
        """Compress and save a single day's events."""
        month_dir = self.archive_dir / event_date.strftime("%Y-%m")
        month_dir.mkdir(parents=True, exist_ok=True)
        
        archive_file = month_dir / f"timeline_{event_date.isoformat()}.jsonl.gz"
        
        with gzip.open(archive_file, 'wt', encoding='utf-8') as f:
            for event in events:
                f.write(json.dumps(event) + '\n')
        
        print(f"[Rotator] Archived {len(events)} events to {archive_file}")


# archival/compressor.py

class MonthlyRollup:
    """Generate monthly summary from archived days."""
    
    def __init__(self, archive_dir: str):
        self.archive_dir = Path(archive_dir)
    
    def generate_month_summary(self, year: int, month: int):
        month_dir = self.archive_dir / f"{year}-{month:02d}"
        if not month_dir.exists():
            return
        
        summary_file = month_dir / "month_summary.json"
        if summary_file.exists():
            return  # Already generated
        
        # Aggregate stats from all days
        total_events = 0
        total_sessions = 0
        all_topics = []
        key_moments = []
        mood_counts = {}
        
        for day_file in sorted(month_dir.glob("timeline_*.jsonl.gz")):
            with gzip.open(day_file, 'rt') as f:
                events = [json.loads(line) for line in f]
            
            total_events += len(events)
            
            for event in events:
                if event.get("event_type") == "session_summary":
                    total_sessions += 1
                    content = event.get("content", {})
                    all_topics.extend(content.get("all_topics", []))
                    key_moments.extend(content.get("key_moments", []))
                    mood = content.get("mood", "unknown")
                    mood_counts[mood] = mood_counts.get(mood, 0) + 1
        
        # Dedupe and count topics
        topic_counts = {}
        for topic in all_topics:
            topic_counts[topic] = topic_counts.get(topic, 0) + 1
        top_topics = sorted(topic_counts.items(), key=lambda x: -x[1])[:20]
        
        summary = {
            "year": year,
            "month": month,
            "total_events": total_events,
            "total_sessions": total_sessions,
            "top_topics": [{"topic": t, "count": c} for t, c in top_topics],
            "mood_distribution": mood_counts,
            "key_moments_count": len(key_moments),
            "key_moments_sample": key_moments[:10],  # Keep first 10 as sample
            "generated_at": datetime.now().isoformat(),
        }
        
        with open(summary_file, 'w') as f:
            json.dump(summary, f, indent=2)
        
        print(f"[Rollup] Generated month summary: {summary_file}")


# archival/retriever.py

class ArchiveRetriever:
    """Query across hot timeline and compressed archives."""
    
    def __init__(self, timeline_path: str, archive_dir: str):
        self.timeline_path = Path(timeline_path)
        self.archive_dir = Path(archive_dir)
    
    def get_events_in_range(
        self, 
        start: datetime, 
        end: datetime
    ) -> list[dict]:
        """Retrieve all events between start and end, across hot and archived."""
        events = []
        
        # Hot timeline
        if self.timeline_path.exists():
            hot_events = load_jsonl(self.timeline_path)
            events.extend([
                e for e in hot_events
                if start <= datetime.fromisoformat(e["timestamp"]) <= end
            ])
        
        # Archives
        current = start.date()
        while current <= end.date():
            month_dir = self.archive_dir / current.strftime("%Y-%m")
            day_file = month_dir / f"timeline_{current.isoformat()}.jsonl.gz"
            
            if day_file.exists():
                with gzip.open(day_file, 'rt') as f:
                    day_events = [json.loads(line) for line in f]
                events.extend([
                    e for e in day_events
                    if start <= datetime.fromisoformat(e["timestamp"]) <= end
                ])
            
            current += timedelta(days=1)
        
        return sorted(events, key=lambda e: e["timestamp"])
    
    def get_recent_sessions(self, n: int = 5) -> list[dict]:
        """Get the N most recent session summaries."""
        # Start from hot, work backwards through archives
        sessions = []
        
        # Hot timeline
        if self.timeline_path.exists():
            hot_events = load_jsonl(self.timeline_path)
            sessions.extend([
                e for e in hot_events
                if e.get("event_type") == "session_summary"
            ])
        
        if len(sessions) >= n:
            return sorted(sessions, key=lambda e: e["timestamp"], reverse=True)[:n]
        
        # Need more from archives
        month_dirs = sorted(self.archive_dir.glob("????-??"), reverse=True)
        
        for month_dir in month_dirs:
            day_files = sorted(month_dir.glob("timeline_*.jsonl.gz"), reverse=True)
            
            for day_file in day_files:
                with gzip.open(day_file, 'rt') as f:
                    day_events = [json.loads(line) for line in f]
                sessions.extend([
                    e for e in day_events
                    if e.get("event_type") == "session_summary"
                ])
                
                if len(sessions) >= n:
                    return sorted(sessions, key=lambda e: e["timestamp"], reverse=True)[:n]
        
        return sorted(sessions, key=lambda e: e["timestamp"], reverse=True)
```

```python
First conversation with this user. No prior history.
Stance: warm introduction, get to know them."""
        
        threads_str = "\n".join(f"  - {t}" for t in self.open_threads) if self.open_threads else "  (none)"
        
        return f"""## Session Continuity

Last talked: {self.time_elapsed} ({self.source})
Topic: {self.last_topic}
What happened: {self.session_summary}
Where we left off: {self.last_exchange_summary}

Open threads:
{threads_str}

Session vibe: {self.mood} (energy: {self.energy:.0%})
Today's sessions: {self.session_count_today} | This week: {self.session_count_week}
Stance: {self.suggested_stance}

---
Continue naturally. You just talked. Don't announce "welcome back" robotically."""


def generate_continuity_packet(
    timeline_path: str,
    archive_dir: str
) -> ContinuityPacket:
    retriever = ArchiveRetriever(timeline_path, archive_dir)
    
    # Get recent session summaries
    recent_sessions = retriever.get_recent_sessions(n=3)
    
    if not recent_sessions:
        return ContinuityPacket.first_meeting()
    
    last_session = recent_sessions[0]
    content = last_session.get("content", {})
    
    last_time = datetime.fromisoformat(last_session["timestamp"])
    elapsed = datetime.now(timezone.utc) - last_time
    
    # Count sessions today and this week
    now = datetime.now(timezone.utc)
    today_start = now.replace(hour=0, minute=0, second=0, microsecond=0)
    week_start = today_start - timedelta(days=now.weekday())
    
    all_today = retriever.get_events_in_range(today_start, now)
    all_week = retriever.get_events_in_range(week_start, now)
    
    sessions_today = len([e for e in all_today if e.get("event_type") == "session_summary"])
    sessions_week = len([e for e in all_week if e.get("event_type") == "session_summary"])
    
    return ContinuityPacket(
        last_interaction=last_time,
        time_elapsed=humanize_delta(elapsed),
        source=last_session.get("source", "unknown"),
        last_topic=content.get("main_topic", "general"),
        session_summary=content.get("session_summary", ""),
        last_exchange_summary=content.get("last_exchange_summary", ""),
        open_threads=content.get("open_threads", []),
        mood=content.get("mood", "neutral"),
        energy=float(content.get("energy_avg", 0.5)),
        suggested_stance=determine_stance(elapsed, content.get("mood", "neutral")),
        session_count_today=sessions_today,
        session_count_week=sessions_week,
    )
```
