```python
            context += "\n"
        
        return context

class SalienceWeightedBlender(ContextBlender):
    """Emphasize most important - for focused mode."""
    
    def blend(self, nodes: List[MemoryNode]) -> str:
        # Calculate importance scores
        scored = [(n, compute_salience(n)) for n in nodes]
        scored.sort(key=lambda x: x[1], reverse=True)
        
        # Progressive detail (more for important, less for peripheral)
        context = ""
        for node, score in scored:
            if score > 0.8:
                detail = node.content  # Full content
            elif score > 0.5:
                detail = node.content[:300]  # Medium
            else:
                detail = node.content[:100]  # Brief mention
            
            context += f"[Relevance: {score:.1f}] {detail}\n\n"
        
        return context

class AssociativeWebBlender(ContextBlender):
    """Show connections - for creative/playful mode."""
    
    def blend(self, nodes: List[MemoryNode]) -> str:
        # Build connection graph
        graph = build_connection_graph(nodes)
        
        # Present as web of associations
        context = "## Connected Memories:\n\n"
        
        # Start with most central node
        central = find_most_connected(graph)
        context += f"**Core**: {central.content[:200]}\n\n"
        
        # Show connections
        for connection in graph.edges_from(central):
            context += f"  ↳ Connected to: {connection.target.content[:150]}\n"
            context += f"     Relationship: {connection.type}\n\n"
        
        return context
```

## Algorithm Blending & Transitions

### Smooth Blending (Not Hard Switches)

```python
class AlgorithmBlender:
    """Blend multiple algorithms based on personality weights."""
    
    def blend_retrieval(
        self,
        query: str,
        personality: PersonalityVector,
        budget: int
    ) -> List[MemoryNode]:
        """Blend multiple retrieval strategies."""
        
        # Get candidates from multiple strategies
        strategies = []
        
        if personality.playfulness > 0.5:
            weight = personality.playfulness
            results = BroadAssociativeRetrieval().search(query, budget)
            strategies.append((results, weight))
        
        if personality.technical_depth > 0.5:
            weight = personality.technical_depth
            results = NarrowPrecisionRetrieval().search(query, budget)
            strategies.append((results, weight))
        
        if personality.warmth > 0.5:
            weight = personality.warmth
            results = EmotionalResonanceRetrieval().search(query, budget)
            strategies.append((results, weight))
        
        # Weighted merge
        return self._weighted_merge(strategies, budget)
    
    def _weighted_merge(
        self, 
        strategies: List[Tuple[List[MemoryNode], float]],
        budget: int
    ) -> List[MemoryNode]:
        """Merge results with personality-based weighting."""
        
        # Collect all unique nodes with weighted scores
        node_scores = {}
        for results, weight in strategies:
            for i, node in enumerate(results):
                # Score = position + strategy weight
                score = (len(results) - i) * weight
                
                if node.node_id in node_scores:
                    node_scores[node.node_id] += score
                else:
                    node_scores[node.node_id] = score
        
        # Sort by blended score
        sorted_nodes = sorted(
            node_scores.items(),
            key=lambda x: x[1],
            reverse=True
        )
        
        # Return top N
        return [get_node(nid) for nid, score in sorted_nodes[:budget]]
```

### State Transitions

```python
class AlgorithmicState:
    """Tracks current algorithmic configuration."""
    
    def __init__(self):
        self.retrieval_strategy = None
        self.context_blender = None
        self.response_composer = None
        self.transition_momentum = 0.7  # Dampening factor
    
    def update(self, target_personality: PersonalityVector):
        """Smoothly transition to new algorithmic state."""
        
        # Select target algorithms
        target_retrieval = select_retrieval_strategy(target_personality)
        target_blender = select_context_blender(target_personality)
        target_composer = select_response_composer(target_personality)
        
        # Check if transition needed
        if type(target_retrieval) != type(self.retrieval_strategy):
            if self._should_transition():
                logger.info(f"Algorithm shift: {type(self.retrieval_strategy).__name__} → {type(target_retrieval).__name__}")
                self.retrieval_strategy = target_retrieval
                self.transition_momentum = 0.3  # Reduce momentum after transition
            else:
                logger.debug("Dampening transition - momentum too low")
                self.transition_momentum += 0.2  # Build momentum
        else:
            self.transition_momentum = min(1.0, self.transition_momentum + 0.1)
        
        # Similar for blender and composer
        self.context_blender = target_blender
        self.response_composer = target_composer
    
    def _should_transition(self) -> bool:
        """Check if enough momentum to change algorithms."""
        return self.transition_momentum > 0.5
```

```python
class TokenBudgetAllocator(ABC):
    """Base class for token budget allocation."""
    
    @abstractmethod
    def allocate(
        self, 
        total_budget: int,
        components: Dict[str, int]
    ) -> Dict[str, int]:
        """Allocate tokens across components."""
        pass

class BalancedAllocator(TokenBudgetAllocator):
    """Even distribution - for general mode."""
    
    def allocate(self, total_budget, components):
        # Equal shares
        n = len(components)
        per_component = total_budget // n
        
        return {k: per_component for k in components}

class MemoryPrioritizedAllocator(TokenBudgetAllocator):
    """Most to memory - for reflective/warm mode."""
    
    def allocate(self, total_budget, components):
        return {
            'system_prompt': int(total_budget * 0.2),
            'state': int(total_budget * 0.1),
            'memory': int(total_budget * 0.6),  # 60% to memory!
            'query': int(total_budget * 0.1)
        }

class PrecisionAllocator(TokenBudgetAllocator):
    """Minimal everything - for focused/direct mode."""
    
    def allocate(self, total_budget, components):
        return {
            'system_prompt': int(total_budget * 0.3),
            'state': int(total_budget * 0.1),
            'memory': int(total_budget * 0.3),  # Less memory
            'query': int(total_budget * 0.3)    # More query space
        }
```

```python
def _pull_context(
    self,
    query: str,
    personality: PersonalityVector,
    budget: int
) -> MemoryContext:
    """Pull context using personality-routed algorithms."""
    
    # Select retrieval strategy
    strategy = select_retrieval_strategy(personality)
    
    # Or blend multiple strategies
    if self.config.blend_algorithms:
        blender = AlgorithmBlender()
        nodes = blender.blend_retrieval(query, personality, budget)

    return MemoryContext(
        nodes=nodes,
        formatted=formatted_context,
        strategy_used=type(strategy).__name__,
        blender_used=type(blender).__name__
    )
```
