**Purpose:** Index external documents (PDFs, markdown, code, etc.) so Luna can search them

**Flow:**

External Doc → Parser → SemanticChunker → Chunk objects → LibraryIndex → DOCUMENT nodes in Memory Matrix

**Chunking Strategy:**
- **Structural**: Parse headings, paragraphs, code blocks, tables
- **Semantic**: Optionally use LLM to detect natural boundaries
- **Size-based**: Merge small chunks (<50 tokens), split large chunks (>1000 tokens)

**Chunk → MemoryNode Mapping:**

Chunk {

    source_id: "mars-college-docs",

```python
"""
Agent Manager
=============

Manages lifecycle of background agents.
Provides registry, scheduling, and error recovery.
"""
import asyncio
import logging
from typing import Dict, Protocol
from datetime import datetime

from .agents import AgentRegistry, AgentState

logger = logging.getLogger(__name__)


class BackgroundAgent(Protocol):
    """Protocol for background agents."""
    
    agent_type: str
    tick_interval: float
    
    async def tick(self) -> None:
        """Execute one tick of agent logic."""
        ...
    
    async def start(self) -> None:
        """Initialize agent."""
        ...
    
    async def stop(self) -> None:
        """Cleanup agent."""
        ...


class AgentManager:
    """
    Manages background agents.
    
    Responsibilities:
    - Register/deregister agents
    - Schedule ticks at appropriate intervals
    - Error recovery (restart crashed agents)
    - Coordinate agent communication
    """
    
    def __init__(self, registry: AgentRegistry, memory):
        self.registry = registry
        self.memory = memory
        self._agents: Dict[str, BackgroundAgent] = {}
        self._tasks: Dict[str, asyncio.Task] = {}
        self._running = False
        
    async def register_agent(self, agent: BackgroundAgent):
        """
        Register a background agent.
        
        Args:
            agent: Agent instance implementing BackgroundAgent protocol
        """
        agent_id = self.registry.register_persistent_agent(
            agent_type=agent.agent_type,
            task=f"{agent.agent_type} background processing",
            tick_interval=agent.tick_interval,
            current_version=0
        )
        
        self._agents[agent_id] = agent
        
        if self._running:
            await self._start_agent(agent_id)
        
        logger.info(f"Registered agent: {agent.agent_type} ({agent_id})")
        return agent_id
    
    async def start(self):
        """Start all registered agents."""
        self._running = True
        
        for agent_id in self._agents:
            await self._start_agent(agent_id)
        
        logger.info(f"AgentManager started with {len(self._agents)} agents")
    
    async def stop(self):
        """Stop all agents gracefully."""
        self._running = False
        
        for agent_id, task in self._tasks.items():
            task.cancel()
            agent = self._agents[agent_id]
            await agent.stop()
        
        await asyncio.gather(*self._tasks.values(), return_exceptions=True)
        self._tasks.clear()
        
        logger.info("AgentManager stopped")
    
    async def _start_agent(self, agent_id: str):
        """Start individual agent tick loop."""
        agent = self._agents[agent_id]
        await agent.start()
        
        task = asyncio.create_task(self._agent_loop(agent_id, agent))
        self._tasks[agent_id] = task
    
    async def _agent_loop(self, agent_id: str, agent: BackgroundAgent):
        """Run agent tick loop with error recovery."""
        while self._running:
            try:
                # Tick
                await agent.tick()
                
                # Update registry
                state = self.registry.get(agent_id)
                if state:
                    state.last_tick = datetime.now()
                    state.last_step += 1
                
            except asyncio.CancelledError:
                break
            except Exception as e:
                logger.error(f"Agent {agent_id} error: {e}", exc_info=True)
                # TODO: Implement restart policy
            
            # Sleep until next tick
            await asyncio.sleep(agent.tick_interval)
```

### 2. Reference Caching

```python
class ReferenceCache:
    """LRU cache for recently loaded sessions"""
    
    def __init__(self, max_size=50):
        self.cache = {}  # reference_file -> content
        self.access_order = deque(maxlen=max_size)
    
    def get(self, reference_file):
        if reference_file in self.cache:
            return self.cache[reference_file]  # ~1ms (cache hit)
        
        content = Path(reference_file).read_text()  # ~15ms (cache miss)
        self.cache[reference_file] = content
        return content
```

**Impact:**
