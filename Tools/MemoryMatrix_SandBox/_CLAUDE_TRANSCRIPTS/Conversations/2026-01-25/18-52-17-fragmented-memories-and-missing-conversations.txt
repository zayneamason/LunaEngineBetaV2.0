## Recommended Fix: Unified Interface

Instead of patching `PersonaAdapter`, create a clean public API on the engine:

```python
# In LunaEngine (engine.py)

async def record_conversation_turn(
    self,
    role: str,  # "user" or "assistant"
    content: str,
    source: str = "text",  # "text", "voice", "api"
    tokens: Optional[int] = None,
) -> None:
    """
    Public API for recording conversation turns.
    
    Handles:
    - Extraction (Scribe → Librarian → Memory Matrix)
    - History storage (HistoryManager)
    - Legacy storage (Matrix store_turn)
    - Context tracking
    
    Args:
        role: Who spoke ("user" or "assistant")
        content: What was said
        source: Input source for logging/debugging
        tokens: Optional token count
    """
    if len(content) < 5:
        return  # Skip trivial messages
        
    # 1. Trigger extraction pipeline
    await self._trigger_extraction(role, content)
    
    # 2. Store in HistoryManager
    history_manager = self.get_actor("history_manager")
    if history_manager and history_manager.is_ready:
        await history_manager.add_turn(
            role=role,
            content=content,
            tokens=tokens or len(content) // 4,
        )
    
    # 3. Store in Matrix (legacy)
    matrix = self.get_actor("matrix")
    if matrix and matrix.is_ready:
        await matrix.store_turn(
            session_id=self.session_id,
            role=role,
            content=content,
            tokens=tokens,
        )
    
    logger.debug(f"Recorded {source} turn: {role} ({len(content)} chars)")
```

if self._engine:
    await self._engine.record_conversation_turn("user", message, source="voice")
