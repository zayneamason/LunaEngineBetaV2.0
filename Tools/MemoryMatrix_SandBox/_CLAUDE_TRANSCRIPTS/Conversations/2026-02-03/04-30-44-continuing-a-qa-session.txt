```jsx
  const [showMetrics, setShowMetrics] = useState(true);

    setAnimating(true);
    setFlowPath([]);
    const path = CONFIG.flows[queryType];

      setTimeout(() => {
        setActiveStep(stepId);
        setFlowPath(prev => [...prev, stepId]);
        if (i === path.length - 1) {
          setTimeout(() => setAnimating(false), 500);
        }
      }, i * 400);
    });
  };

  const isInPath = (stepId) => flowPath.includes(stepId);
  const isActive = (stepId) => activeStep === stepId;

  const getStatusColor = (status) => {
    switch(status) {
      case 'stable': return '#22c55e';
      case 'needs_work': return '#f59e0b';
      case 'broken': return '#ef4444';
      case 'planned': return '#6366f1';
      default: return '#6b7280';
    }
  };

  const getStepStyle = (step) => {
    const base = {
      position: 'absolute',
      left: step.x + 200,
      top: step.y,
      width: 160,
      padding: '12px 16px',
      borderRadius: 8,
      textAlign: 'center',
      cursor: 'pointer',
      transition: 'all 0.3s ease',
      fontFamily: "'JetBrains Mono', monospace",
      fontSize: 12,
    };

    if (isActive(step.id)) {
      return {
        ...base,
        background: 'linear-gradient(135deg, #a855f7 0%, #7c3aed 100%)',
        color: '#fff',
        boxShadow: '0 0 30px rgba(168, 85, 247, 0.6)',
        transform: 'scale(1.08)',
        border: '2px solid #c084fc',
      };
    }
    
    if (isInPath(step.id)) {
      return {
        ...base,
        background: 'rgba(168, 85, 247, 0.15)',
        color: '#e9d5ff',
        border: `2px solid ${getStatusColor(step.status)}`,
        boxShadow: `0 2px 10px ${getStatusColor(step.status)}33`,
      };
    }

    return {
      ...base,
      background: 'rgba(30, 30, 40, 0.8)',
      color: '#a1a1aa',
      border: '1px solid rgba(100, 100, 120, 0.3)',
    };
  };

  const renderConnections = () => (
    <svg style={{ position: 'absolute', top: 0, left: 0, width: '100%', height: '100%', pointerEvents: 'none' }}>

        <marker id="arrow" markerWidth="10" markerHeight="7" refX="9" refY="3.5" orient="auto">
          <polygon points="0 0, 10 3.5, 0 7" fill="#6b7280" />
        </marker>

        const toX = toStep.x + 280;
        const toY = toStep.y;

          <path
            key={i}
            d={pathD}

            </div>

      {/* Controls */}

              padding: '8px 16px',
              borderRadius: 6,
              border: 'none',
              cursor: 'pointer',
              fontFamily: "'JetBrains Mono', monospace",
              fontSize: 11,

              color: queryType === type ? '#fff' : '#a1a1aa',
            }}
          >
            {type.toUpperCase()}
            <span style={{ display: 'block', fontSize: 9, opacity: 0.7 }}>
              {CONFIG.queryExamples[type]}
            </span>
          </button>
        ))}
        <button
          onClick={runAnimation}
          disabled={animating}
          style={{
            padding: '8px 20px',
            borderRadius: 6,
            border: 'none',
            cursor: animating ? 'wait' : 'pointer',
            background: '#22c55e',
            color: '#fff',
            fontWeight: 600,
            fontSize: 12,

        >
          {animating ? '● Running...' : '▶ Run Flow'}

        position: 'relative',
        width: 560,
        height: 880,

      }}>
        {renderConnections()}
        
        {/* Path Labels */}

          position: 'absolute', left: 80, top: 400,
          color: queryType === 'simple' ? '#22c55e' : '#4b5563',
          fontSize: 11, fontWeight: 600, fontFamily: "'JetBrains Mono', monospace",

              <div style={{ fontSize: 10, opacity: 0.7 }}>{step.desc}</div>
            )}
            {step.status && step.status !== 'stable' && (

                position: 'absolute', top: -6, right: -6,

                background: getStatusColor(step.status),
              }} />
            )}
          </div>
        ))}

        {/* Threshold Indicator */}
        <div style={{
          position: 'absolute', left: 380, top: 335,
          background: 'rgba(30, 30, 40, 0.9)',
          padding: '8px 12px', borderRadius: 6,
          border: '1px solid rgba(100, 100, 120, 0.3)',
          fontFamily: "'JetBrains Mono', monospace", fontSize: 10,

          <div style={{ color: '#e9d5ff', fontWeight: 500, marginBottom: 4 }}>Threshold</div>

          <div><span style={{ color: '#f59e0b' }}>&gt; {CONFIG.complexityThreshold}</span> → Delegate</div>
        </div>

      {/* Info Panel */}
      {activeStep && CONFIG.steps[activeStep] && (
        <div style={{
          maxWidth: 560, margin: '24px auto 0', padding: 20,
```

## **The Core Difference: WHO is the LLM?**

### **MCP Path (Working)**
```
User → Claude (THE BRAIN) → decides to search → MCP tool → Memory Matrix → Claude synthesizes → Response
```

**Claude IS the intelligence.** MCP just gives Claude *tools* to access Luna's memories. Claude:
- Already has a massive context window
- Understands intent and reasoning
- Decides WHEN to call `luna_smart_fetch`
- Synthesizes results naturally into responses

**The MCP tools are thin** — they just proxy to the Engine API for search/add/read operations. Claude does the heavy lifting.

---

### **Direct API Path (Struggling)**
```
User → Director → Entity Detect + Memory Fetch + Mood Analysis → BUILD MASSIVE PROMPT → Qwen + LoRA → Response
```

**Qwen is just a completion engine.** It has no persistent intelligence. The Director must:
- Build EVERYTHING into the prompt
- Include DNA, Experience, Mood, Entities, Memories, History, Voice Examples
- Hope Qwen "understands" and generates appropriately

**The problem:** Qwen sees voice examples like `"yo, that's kinda wild"` and **copies them verbatim** instead of absorbing the style. It pattern-matches, not comprehends.

---

## **Why MCP "Just Works"**

| Aspect | MCP | Direct API |
|--------|-----|------------|
| **LLM** | Claude (Opus/Sonnet) | Qwen 7B + LoRA |
| **Intelligence** | Built into the model | Must be in the prompt |
| **Memory access** | Tool call when needed | Pre-fetched, stuffed in prompt |
| **Voice** | Claude reads style guide, synthesizes | Qwen copies examples |
| **Context** | 200K tokens available | ~4K effective |
| **Reasoning** | Deep understanding | Surface pattern matching |

---

## **The Fix**

Don't put voice **examples** in the prompt — **train Luna's voice INTO the LoRA weights**.

The model should *be* Luna, not *imitate* Luna from examples.

That's exactly what Persona Forge is for — take the 415 markdown files + Memory Matrix data, create fine-tuning examples, and bake Luna's personality into the weights themselves.

## Reasoning Depth: What It Actually Refers To

### **1. Literal Layer Count**

```
Claude Opus:  ~100+ transformer layers
Qwen 7B:      32 transformer layers
```

Each layer is a chance for the model to:
- Attend to different parts of the context
- Transform representations
- Build more abstract understanding

More layers = more sequential computation = more "thinking steps" before outputting.

---

### **2. Attention Patterns (Multi-Hop Reasoning)**

In self-attention, every token can attend to every other token. But the DEPTH of reasoning comes from **chaining** these attention hops across layers:

```
Layer 1:  "marzipan" attends to "who is"
          → understands this is a question about a person

Layer 5:  "marzipan" attends to "luna_smart_fetch" tool definition
          → starts connecting "person query" to "memory tool"

Layer 20: Representation now encodes "this needs a memory lookup"
          → activates tool-use circuits

Layer 50: Builds the actual tool call structure
          → formats parameters correctly

Layer 80: Refines output, checks coherence
          → ensures response makes sense
```

**Qwen with 32 layers** has fewer opportunities for these transformations. It's like trying to solve a multi-step problem but only getting 32 "thinking moments" vs 100+.

---

### **3. Circuit Depth (Mechanistic Interpretability View)**

Research from Anthropic's interpretability team shows that:

- **Shallow circuits** (early layers): Detect simple patterns — "is this a question?", "is there a name?"
- **Mid circuits** (middle layers): Combine features — "question about a person + tool available"
- **Deep circuits** (late layers): Abstract reasoning — "I should use this tool because..."


Larger/deeper models show **emergent capabilities** that smaller ones don't:

Each transformer layer has:
- **Attention** (dynamic, context-dependent)

Qwen has less "world knowledge" baked in, so it relies more heavily on what's in the prompt.
