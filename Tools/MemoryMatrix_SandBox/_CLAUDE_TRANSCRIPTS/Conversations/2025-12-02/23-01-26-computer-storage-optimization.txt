            chosen_action = self.select_action(possible_actions)
            
            # 4. ACT - Do the thing (or choose to wait)
            if chosen_action and chosen_action.value > THRESHOLD:
                outcome = await self.executor.execute(chosen_action)
                
                # 5. REFLECT - Did it work?
                self.learn_from_outcome(chosen_action, outcome)
            
            await asyncio.sleep(self.tick_rate)  # Don't spin
```

def generate_actions(self, goals: list[Goal]) -> list[PossibleAction]:
    actions = []
    
    for goal in goals:
        # What could I do toward this goal given current world model?

                actions.append(PossibleAction(
                    type="offer_help",
                    content=self.generate_help_offer(),

                    confidence=0.8,
                    intrusiveness=0.6,  # Might interrupt flow
                ))

                actions.append(PossibleAction(
                    type="surface_context",
                    content=self.retrieve_relevant_memory(),
                    goal=goal,
                    confidence=0.7,
                    intrusiveness=0.3,  # Low-key helpful
                ))
        
        if goal.objective == "Protect focus when deep in work":
            if self.world_model.focus_state == "deep":
                actions.append(PossibleAction(
                    type="suppress_notifications",
                    goal=goal,
                    confidence=0.9,
                    intrusiveness=0.0,  # Silent action
                ))
    
    # Always include "wait and observe" as an option
    actions.append(PossibleAction(
        type="wait",
        goal=None,
        confidence=1.0,
        intrusiveness=0.0,
    ))
    
    return actions

def select_action(self, actions: list[PossibleAction]) -> PossibleAction:
    """Pick the best action considering goals AND context"""
    
    for action in sorted(actions, key=lambda a: a.score()):
        # Check: Is this worth the interruption cost?
        if action.intrusiveness > self.world_model.interrupt_tolerance:
            continue  # Skip intrusive actions when Ahab needs focus
            
        # Check: Am I confident enough to act?
        if action.confidence < self.world_model.action_threshold:
            continue
            
        return action
    
    return WAIT_ACTION  # Default: observe more before acting
```
