### Priority & Conflict Resolution

When multiple states compete (e.g., I'm both thinking AND responding to text gestures):

```typescript
class OrbStateManager {
  private stateQueue: Array<{ state: OrbState, timestamp: number }> = [];
  
  // Add new state
  push(state: OrbState) {
    this.stateQueue.push({ state, timestamp: Date.now() });
    this.resolve();
  }
  
  // Resolve conflicts
  private resolve() {
    // Priority order:
    // 1. System errors (highest)
    // 2. System events (voice, processing)
    // 3. Text gestures (explicit Luna expressions)
    // 4. Idle state (lowest)
    
    const sorted = this.stateQueue
      .filter(s => !s.state.duration || Date.now() - s.timestamp < s.state.duration)
      .sort((a, b) => b.state.priority - a.state.priority);
    
    return this.blend(sorted[0]?.state, sorted[1]?.state);
  }
  
  // Blend two states for smooth transitions
  private blend(primary: OrbState, secondary?: OrbState): OrbState {
    if (!secondary) return primary;
    
    // Blend rules:
    // - Animation from primary
    // - Color can mix if compatible
    // - Brightness/size averaged
    // - Position interpolated
    
    return {
      ...primary,
      brightness: (primary.brightness + secondary.brightness) / 2,
      // ... blend logic
    };
  }
}
```
