```python
    }[objective]
    
    latency_score = max(0, 1 - (eval_results.avg_latency_ms / 2000))
    
    return (
        weights["memory"] * eval_results.memory_recall_score +
        weights["context"] * eval_results.context_retention_score +
        weights["routing"] * eval_results.routing_score +
        weights["latency"] * latency_score
    )
```
