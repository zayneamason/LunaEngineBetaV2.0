```python
def apply_offline_decay(cluster: Cluster, offline_seconds: int):
    """Decay lock-in logarithmically during offline time."""
    
    # Decay constants by state
    lambda_decay = {
        'drifting': 0.001,      # Fast decay
        'fluid': 0.0005,        # Medium decay
        'settled': 0.0001,      # Slow decay
        'crystallized': 0.00001 # Negligible decay
    }
    
    decay = lambda_decay[cluster.state]
    cluster.lock_in *= math.exp(-decay * offline_seconds)
    
    # Crystallized memories stay solid even after months
    # Drifting memories vanish in days
```
