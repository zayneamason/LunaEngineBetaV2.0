```python
# voice/tts/piper.py
import logging
from pathlib import Path
from typing import List, Optional, AsyncIterator
import asyncio

from piper import PiperVoice
from .provider import TTSProvider
from voice.conversation.state import AudioBuffer, VoiceInfo

logger = logging.getLogger(__name__)

class PiperTTS(TTSProvider):
    """
    Piper TTS - Fast local neural TTS.
    
    Setup:
    1. pip install piper-tts
    2. Download voices from: https://github.com/rhasspy/piper/releases
    3. Put .onnx files in ~/.local/share/piper/voices/
    """
    
    def __init__(self, model_path: Optional[str] = None, voice: str = "en_US-lessac-medium"):
        """
        Initialize Piper TTS.
        
        Args:
            model_path: Path to .onnx model file
            voice: Voice name (if model_path not provided, downloads automatically)
        """
        self.voice_name = voice
        self.model_path = model_path
        self._voice: Optional[PiperVoice] = None
        self._available = False
        
        # Try to load voice
        try:
            if model_path and Path(model_path).exists():
                self._voice = PiperVoice.load(model_path)
            else:
                # Auto-download voice (piper handles this)
                from piper.download import ensure_voice_exists, find_voice
                voice_path = ensure_voice_exists(voice)
                self._voice = PiperVoice.load(voice_path)
            
            self._available = True
            logger.info(f"Piper TTS initialized: {voice}")
        except Exception as e:
            logger.warning(f"Piper TTS unavailable: {e}")
    
    def is_available(self) -> bool:
        return self._available
    
    def get_name(self) -> str:
        return f"Piper ({self.voice_name})"
    
    async def synthesize(self, text: str, voice: Optional[str] = None) -> AudioBuffer:
        """Generate complete audio for text."""
        if not self._voice:
            raise RuntimeError("Piper voice not loaded")
        
        # Run in thread pool (piper is synchronous)
        loop = asyncio.get_event_loop()
        audio_bytes = await loop.run_in_executor(
            None,
            lambda: bytes(self._voice.synthesize_stream_raw(text))
        )
        
        return AudioBuffer(
            data=audio_bytes,
            sample_rate=self._voice.config.sample_rate,
            channels=1,
            format="pcm_s16le"
        )
    
    async def stream_synthesize(self, text: str, voice: Optional[str] = None) -> AsyncIterator[AudioBuffer]:
        """Stream audio chunks as they're generated."""
        if not self._voice:
            raise RuntimeError("Piper voice not loaded")
        
        loop = asyncio.get_event_loop()
        
        # Piper generates in chunks
        for chunk in self._voice.synthesize_stream_raw(text):
            # Yield from executor
            audio_bytes = await loop.run_in_executor(None, lambda: bytes(chunk))
            
            yield AudioBuffer(
                data=audio_bytes,
                sample_rate=self._voice.config.sample_rate,
                channels=1,
                format="pcm_s16le"
            )
    
    def get_voices(self) -> List[VoiceInfo]:
        """List available Piper voices."""
        # Common Piper voices for English
        return [
            VoiceInfo(id="en_US-lessac-medium", name="Lessac (US Male)", language="en-US"),
            VoiceInfo(id="en_US-amy-medium", name="Amy (US Female)", language="en-US"),
            VoiceInfo(id="en_GB-alba-medium", name="Alba (UK Female)", language="en-GB"),
            VoiceInfo(id="en_GB-danny-low", name="Danny (UK Male)", language="en-GB"),
        ]
```

# voice/cli.py
import asyncio
import argparse
from .backend import VoiceBackend
from .tts.manager import TTSProviderType
from .stt.manager import STTProviderType

async def main():
    parser = argparse.ArgumentParser(description="Luna Voice Interface")
    parser.add_argument("--tts", choices=["apple", "edge", "piper"],

    parser.add_argument("--voice", help="TTS voice ID")

    # Map string to enum
    tts_map = {
        "apple": TTSProviderType.APPLE,
        "edge": TTSProviderType.EDGE,
        "piper": TTSProviderType.PIPER,
    }
    
    stt_map = {
        "mlx_whisper": STTProviderType.MLX_WHISPER,
        "apple": STTProviderType.APPLE,
        "whisper": STTProviderType.WHISPER,
    }
    
    backend = VoiceBackend(
        stt_provider=stt_map[args.stt],
        tts_provider=tts_map[args.tts],
        tts_voice=args.voice or "en-US-AriaNeural",  # Default Edge voice
        hands_free=args.hands_free
    )
    
    await backend.start()
    
    print("""
Luna Voice Interface
--------------------
Mode: Push-to-Talk
- Hold SPACEBAR to speak
- Release to process
- Ctrl-C to exit
    """)
    
    # Push-to-talk loop
    import keyboard
    
    try:
        while True:
            if keyboard.is_pressed('space'):
                if not backend._recording:
                    await backend.start_listening()
            else:
                if backend._recording:
                    await backend.handle_push_to_talk()
            
            await asyncio.sleep(0.01)
    
    except KeyboardInterrupt:
        print("\nShutting down...")
    finally:
        await backend.stop()

if __name__ == "__main__":
    asyncio.run(main())

class CollisionDetector:
    """Detects user speech during Luna's response."""
    
    def __init__(self, sensitivity: float = 0.5):
        self.vad = SileroVAD()  # Use better VAD
        self.sensitivity = sensitivity
        self.cooldown_ms = 500  # Ignore first 500ms
        self.min_duration_ms = 300  # Must speak for 300ms
    
    async def monitor_during_playback(self, playback_task):
        """
        Monitor mic while Luna is speaking.
        Cancel playback if user interrupts.
        """
        start_time = time.time()
        speech_start = None
        
        while not playback_task.done():
            chunk = self.audio_capture.get_chunk(timeout=0.05)
            if not chunk:
                continue
            
            elapsed_ms = (time.time() - start_time) * 1000
            if elapsed_ms < self.cooldown_ms:
                continue  # Ignore initial period
            
            is_speech = self.vad.detect(chunk)
            
            if is_speech:
                if speech_start is None:
                    speech_start = time.time()
                
                duration_ms = (time.time() - speech_start) * 1000
                if duration_ms > self.min_duration_ms:
                    # Real interrupt!
                    playback_task.cancel()
                    return True
            else:
                speech_start = None
        
        return False
```

1. **Hard stop** (simplest)

   - Start listening

   - Detect sentence end in response text

   - Start listening

3. **Fade out** (smoothest)
   - Quick volume fade (200ms)
   - Stop playback

## What Each Component Does

### **AudioCapture** (`audio/capture.py`)
- Opens microphone via sounddevice
- Records audio in chunks (1024 samples)
- Provides queue-based buffering
- Returns raw PCM audio bytes

### **AudioPlayback** (`audio/playback.py`)
- Plays audio to speakers via sounddevice
- Handles multiple audio formats (WAV, PCM)
- Async playback with cancellation support

### **STTManager** (`stt/manager.py`)
- Manages 3 STT providers with fallback:
  1. MLX Whisper (primary - fast, local, Apple Silicon)
  2. Apple STT (fallback - native macOS)
  3. Whisper (fallback - standard)
- Automatic provider switching on failure
- Returns `TranscriptionResult` with text + confidence

### **TTSManager** (`tts/manager.py`)
- Manages 3 TTS providers with fallback:
  1. Kokoro (primary - neural, Docker)
  2. Edge TTS (fallback - online, Microsoft)
  3. Apple TTS (fallback - native macOS)
- Voice selection per provider
- Female-only voice filtering for Luna
- Prosody control (rate, pitch, volume)
- Returns `AudioBuffer` with synthesized speech

### **ConversationManager** (`conversation/manager.py`)
- Tracks conversation state (idle/listening/processing/speaking)
- Turn-taking protocol (user/assistant)
- Sliding window history (default 10 turns)
- Session management integration
- Persists to Memory Matrix via SessionManager

### **ProsodyMapper** (`prosody.py`)
- Takes personality vector from PersonaCore
- Maps traits to voice parameters:
  - Warmth → pitch (+10% to -10%)
  - Energy → rate (+15% to -15%)
  - Confidence → volume (+5% to -5%)
- Outputs prosody params for TTS

### **WakeWordDetector** (`wakeword.py`)
- Continuously listens in background
- Processes 2-second audio chunks
- Uses STT to detect "Hey Luna" or "Luna"
- Energy threshold to avoid processing silence
- Cooldown period after activation (2 seconds)

### **VoiceBackend** (`backend.py`)
**Main orchestrator that connects everything:**

**Modes:**
- Push-to-talk: Hold key → speak → release → process
- Hands-free: VAD auto-detects speech start/end

**Flow:**
1. AudioCapture records speech
2. VAD detects speech boundaries (hands-free mode)
3. STTManager transcribes audio
4. HubClient sends to PersonaCore
5. PersonaCore returns personality-enriched response
6. ProsodyMapper modulates voice based on personality
7. TTSManager synthesizes speech with prosody
8. AudioPlayback plays response

**Features:**
- Component health monitoring
- Graceful degradation (provider fallback)
- Status reporting to Hub
- Callback system (on_speech_start, on_speech_end, on_response)
- Session persistence

### **HubClient** (`hub_client.py`)
**HTTP client to Eclissi Hub:**
- Connects to Hub daemon (http://127.0.0.1:8882)
- Sends voice queries to `/voice/process` endpoint
- Receives `HubResponse`:
  - `response` - Luna's text response
  - `memory_context` - Retrieved memories
  - `state` - Personality vector + consciousness state
- Health checks Hub availability
- Graceful degradation when Hub offline

            persona_core: PersonaCore instance from runtime
        """
        from persona.core import PersonaCore
        
        if not isinstance(persona_core, PersonaCore):
            raise TypeError(f"Expected PersonaCore, got {type(persona_core)}")
        
        self.persona = persona_core
        self._available = True
        
        logger.info("PersonaCore adapter initialized (direct integration)")
    
    def is_available(self) -> bool:
        """Check if PersonaCore is available."""
        return self._available and self.persona is not None
    
    async def connect(self) -> bool:
        """
        Connect to PersonaCore (compatibility with HubClient API).
        
        Since we're using direct integration, this always succeeds.
        """
        return self.is_available()
    
    async def process_message(
        self,
        message: str,
        interface: str = "voice",
        session_id: Optional[str] = None,
        generate_response: bool = True
    ) -> Optional[VoiceResponse]:
        """
        Process a message through PersonaCore.
        
        Args:
            message: User message
            interface: Interface type (voice | desktop)
            session_id: Optional session ID
            generate_response: If True, generates LLM response
        
        Returns:
            VoiceResponse with Luna's personality-enriched response
        """
        if not self.is_available():
            logger.error("PersonaCore not available")
            return None
        
        try:
            # Process query through PersonaCore
            # Budget: "balanced" for voice (not too heavy, not too light)
            result = self.persona.process_query(
                query=message,
                budget="balanced"  # Voice uses balanced budget
            )
            
            # Extract personality state
            personality_state = {
                "personality": result.personality.to_dict() if result.personality else {},
                "attention": {},  # TODO: Get from consciousness system if needed
                "coherence": {},  # TODO: Get from consciousness system if needed
            }
            
            # Generate response if requested
            if generate_response:
                # TODO: This needs Director LLM integration
                # For now, return a placeholder or integrate with current LLM
                # When Director is ready, this calls Director.generate()
                
                # Temporary: Use Claude API via existing pattern
                # Or return result.generated_response if Director already wired
                response_text = result.generated_response if hasattr(result, 'generated_response') else "Response generation not yet implemented"
            else:
                response_text = ""
            
            return VoiceResponse(
                response=response_text,
                memory_context=result.memory_context or [],
                state=personality_state,
                debug={
                    "budget_used": result.budget_used if hasattr(result, 'budget_used') else 0,
                    "memories_retrieved": len(result.memory_context) if result.memory_context else 0,
                }
            )
        
        except Exception as e:
            logger.error(f"PersonaCore processing failed: {e}")
            return None
    
    async def close(self):
        """Close adapter (compatibility with HubClient API)."""
        # No cleanup needed for direct integration
        pass
```
