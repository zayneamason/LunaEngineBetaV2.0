### Hub-Centric Conversation Buffer

```python
# In Hub - single source of truth
class ConversationBuffer:
    """Hub-managed conversation history."""
    
    def __init__(self, max_turns: int = 20, max_tokens: int = 4000):
        self.max_turns = max_turns
        self.max_tokens = max_tokens
        self.turns: List[Dict] = []
        self.session_id: str = None
        
    def add_turn(self, role: str, content: str, interface: str):
        """Add turn from any interface."""
        self.turns.append({
            "role": role,
            "content": content,
            "interface": interface,
            "timestamp": datetime.now().isoformat()
        })
        self._trim()
        
    def get_for_llm(self, max_tokens: int = None) -> List[Dict]:
        """Get history formatted for LLM, respecting token budget."""
        budget = max_tokens or self.max_tokens
        result = []
        tokens = 0
        
        for turn in reversed(self.turns):
            turn_tokens = len(turn["content"]) // 4
            if tokens + turn_tokens > budget:
                break
            result.insert(0, {"role": turn["role"], "content": turn["content"]})
            tokens += turn_tokens
            
        return result
```
