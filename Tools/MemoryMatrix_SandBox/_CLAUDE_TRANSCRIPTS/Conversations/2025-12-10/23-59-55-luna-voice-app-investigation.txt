### Agent Task Classification

```python
class AgentRouter:
    """Routes tasks to the most appropriate agent"""
    
    def __init__(self, agents: Dict[str, Agent]):
        self.agents = agents
        self.classifier = TaskClassifier()  # Could be MLX-based
    
    async def route(self, task: Task, context: Context) -> AgentResult:
        # 1. Classify task type
        task_type = self.classifier.classify(task)
        # → "conversation" | "tool_use" | "coding" | "research" | "file_op"
        
        # 2. Check constraints
        constraints = self.get_constraints(context)
        # → {offline: True, latency_budget_ms: 500, cost_budget: 0.0}
        
        # 3. Select agent
        agent = self.select_agent(task_type, constraints)
        
        # 4. Execute
        result = await agent.execute(task, context)
        
        # 5. Log to Memory Matrix (all agents contribute to shared memory)
        await self.log_agent_action(agent, task, result)
        
        return result
    
    def select_agent(self, task_type: str, constraints: Constraints) -> Agent:
        candidates = self.agents_for_type[task_type]
        
        # Filter by constraints
        if constraints.offline:
            candidates = [a for a in candidates if a.supports_offline]
        
        if constraints.latency_budget_ms < 500:
            candidates = [a for a in candidates if a.avg_latency_ms < constraints.latency_budget_ms]
        
        # Return highest capability that fits constraints
        return max(candidates, key=lambda a: a.capability_score)
```

## 4. Online/Offline LLM Strategy

### The LLM Router

```python
class LLMStrategyRouter:
    """
    Routes LLM requests based on:
    - Network availability
    - Task complexity
    - Latency requirements
    - Cost budget
    - Privacy sensitivity
    """
    
    def __init__(self):
        self.providers = {
            "claude": ClaudeProvider(),      # Online, highest quality
            "mlx": MLXProvider(),            # Local, Apple Silicon
            "smollm": SmolLMProvider(),      # Local, lightweight
        }
        self.network_monitor = NetworkMonitor()
        self.cost_tracker = CostTracker()
    
    async def route(self, request: LLMRequest) -> LLMResponse:
        # 1. Check network
        is_online = await self.network_monitor.is_connected()
        
        # 2. Classify task complexity
        complexity = self.classify_complexity(request)
        # → "simple" | "moderate" | "complex"
        
        # 3. Check constraints
        constraints = request.constraints
        
        # 4. Select provider
        provider = self.select_provider(
            is_online=is_online,
            complexity=complexity,
            constraints=constraints
        )
        
        # 5. Execute with fallback
        try:
            return await provider.generate(request)
        except ProviderError:
            return await self.fallback(request, exclude=[provider])
    
    def select_provider(
        self,
        is_online: bool,
        complexity: str,
        constraints: LLMConstraints
    ) -> LLMProvider:
        
        # Offline mode: local only
        if not is_online:
            if complexity == "complex":
                return self.providers["mlx"]      # Best local model
            else:
                return self.providers["smollm"]   # Fastest local
        
        # Online mode: route by complexity and constraints
        if constraints.privacy_sensitive:
            return self.providers["mlx"]          # Keep data local
        
        if constraints.max_latency_ms < 200:
            return self.providers["smollm"]       # Fastest
        
        if complexity == "complex" or constraints.quality == "high":
            return self.providers["claude"]       # Best quality
        
        if self.cost_tracker.daily_spend > constraints.daily_budget * 0.8:
            return self.providers["mlx"]          # Save money
        
        # Default: Claude for quality
        return self.providers["claude"]
```

# LunarBehaviour tracks personality over time

class PersonalityTracker:
    """Monitors personality evolution and prevents unhealthy drift"""
    
    def __init__(self, kernel: Kernel):
        self.kernel = kernel
        self.history = []  # Rolling window of personality vectors
    
    def record_personality(self, personality: PersonalityVector):
        self.history.append({
            "timestamp": datetime.now(),
            "vector": personality,
        })
        
        # Check for drift
        self.check_drift()
    
    def check_drift(self):
        if len(self.history) < 100:
            return
        
        recent = self.history[-20:]
        baseline = self.history[:20]
        
        for trait in PersonalityVector.traits:
            recent_avg = mean([h["vector"][trait] for h in recent])
            baseline_avg = mean([h["vector"][trait] for h in baseline])
            
            drift = abs(recent_avg - baseline_avg)
            
            if drift > 0.3:  # Significant drift
                self.alert_drift(trait, baseline_avg, recent_avg)
    
    def alert_drift(self, trait: str, old: float, new: float):
        """
        Personality drifting significantly.
        Could mean:
        - Natural evolution (user interaction style changed)
        - Problematic drift (needs calibration)
        """
        
        # Log for review
        logging.warning(f"Personality drift: {trait} {old:.2f} → {new:.2f}")
        
        # Create kernel update proposal
        if self.kernel.allows_evolution(trait):
            proposal = KernelUpdateProposal(
                trait=trait,
                old_value=old,
                new_value=new,
                evidence=self.history[-50:],
                requires_review=True
            )
            self.kernel.propose_update(proposal)

class DomainExpertiseTracker:

# Optimization strategies for scale:
class ScaledMemoryMatrix:
    def __init__(self):
        self.hot_cache = LRUCache(1000)      # Recent/frequent nodes
        self.graph_cache = GraphCache()       # Precomputed subgraphs
        self.archive_threshold_days = 90     # Move old nodes to archive
    
    async def search(self, query: str) -> List[SearchResult]:
        # Check hot cache first
        cache_key = hash(query)
        if cached := self.hot_cache.get(cache_key):
            return cached
        
        # FTS5 search with recency boost
        results = await self.fts5_search(query)
        
        # Graph expand only top results (expensive)
        expanded = await self.graph_expand(results[:5])
        
        # Cache and return
        self.hot_cache.set(cache_key, expanded)
        return expanded
