### **Update Flow:**

```javascript
// END OF EACH SESSION
updateUserVirtues = (sessionData) => {
  
  // Calculate virtue shifts from session
  shifts = analyzeSessionBehavior(sessionData);
  
  // Apply with dampening (slow accumulation)
  dampening = 0.15; // 15% of shift actually applied
  
  Object.keys(shifts).forEach(virtue => {
    currentValue = userCharacter.virtues[virtue];
    shift = shifts[virtue] * dampening;
    newValue = Math.max(0, Math.min(1, currentValue + shift));
    userCharacter.virtues[virtue] = newValue;
  });
  
  // Save to localStorage
  localStorage.setItem('userCharacter', JSON.stringify(userCharacter));
  
  // Log for analytics
  console.log('Virtue Update:', shifts);
}
```
