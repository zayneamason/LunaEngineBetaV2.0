import socket
import json

SOCKET_PATH = "/tmp/luna/conversation.sock"

class ConversationLogger:
    def __init__(self):
        self.sock = None
        self._connect()
    
    def _connect(self):
        try:
            self.sock = socket.socket(socket.AF_UNIX, socket.SOCK_STREAM)
            self.sock.connect(SOCKET_PATH)
        except:
            self.sock = None  # Fallback mode
    
    def log_turn(self, role: str, content: str, session_id: str):
        if not self.sock:
            return False
        
        payload = json.dumps({
            "type": "turn",
            "role": role,        # "user" | "assistant"
            "content": content,
            "session_id": session_id,
            "timestamp": time.time()
        }) + "\n"
        
        try:
            self.sock.sendall(payload.encode())
            return True
        except:
            self._connect()  # Try reconnect
            return False

# Hook into MCP tool dispatch
conversation_logger = ConversationLogger()

def on_tool_call(tool_name: str, params: dict, context: dict):
    # Log the user message that triggered this
    if "user_message" in context:
        conversation_logger.log_turn("user", context["user_message"], context["session_id"])
    
    # Execute tool...
    result = execute_tool(tool_name, params)
    
    # Log assistant response
    if "assistant_response" in context:
        conversation_logger.log_turn("assistant", context["assistant_response"], context["session_id"])

# In Hub - reads continuously
import asyncio
import socket

class ConversationReceiver:
    def __init__(self, socket_path: str, chunker: Chunker, ben: BenjaminFranklin):
        self.socket_path = socket_path
        self.chunker = chunker
        self.ben = ben
    
    async def listen(self):
        # Create socket
        server = socket.socket(socket.AF_UNIX, socket.SOCK_STREAM)
        
        # Clean up old socket
        if os.path.exists(self.socket_path):
            os.unlink(self.socket_path)
        
        server.bind(self.socket_path)
        server.listen(1)
        server.setblocking(False)
        
        log.info(f"🔌 Listening on {self.socket_path}")
        
        while True:
            conn, _ = await asyncio.get_event_loop().sock_accept(server)
            asyncio.create_task(self._handle_connection(conn))
    
    async def _handle_connection(self, conn):
        buffer = ""
        while True:
            try:
                data = await asyncio.get_event_loop().sock_recv(conn, 4096)
                if not data:
                    break
                
                buffer += data.decode()
                
                # Process complete lines
                while "\n" in buffer:
                    line, buffer = buffer.split("\n", 1)
                    await self._process_turn(json.loads(line))
            except:
                break
        conn.close()
    
    async def _process_turn(self, turn: dict):
        """Feed turn to chunker → Ben pipeline."""
        chunk = self.chunker.process(
            content=turn["content"],
            speaker="ahab" if turn["role"] == "user" else "luna",
            session_id=turn["session_id"],
            timestamp=turn["timestamp"]
        )
        
        stack = self.context_window.build(chunk)
        
        if self.ben.config.enabled:
            extractions = self.ben.on_message(stack)
            log.debug(f"📜 Ben extracted {len(extractions)} from {turn['role']} turn")

class ConversationLogger:
    def log_turn(self, role: str, content: str, session_id: str):
        if self.sock and self._send(payload):
            return True
        
        # Socket failed
        log.warning("⚠️ Conversation socket unavailable - degraded mode")
        self._schedule_reconnect()
        
        # Don't lose the data - buffer locally
        self._buffer_locally(payload)
        return False
    
    def _buffer_locally(self, payload: str):
        """Write to local file as backup."""
        with open("/tmp/luna/buffer.jsonl", "a") as f:
            f.write(payload)
    
    def _on_reconnect(self):
        """Flush buffer when socket comes back."""
        if os.path.exists("/tmp/luna/buffer.jsonl"):
            with open("/tmp/luna/buffer.jsonl") as f:
                for line in f:
                    self.sock.sendall(line.encode())
            os.unlink("/tmp/luna/buffer.jsonl")
