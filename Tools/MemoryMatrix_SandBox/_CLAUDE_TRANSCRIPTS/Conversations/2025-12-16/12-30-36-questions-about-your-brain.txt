        self.janitor.health_check()

class BenjaminFranklin(SmolAgent):
    """

        Called when new message arrives.
        Returns list of extractions.

        extractions = []
        
        # Step 1: Pattern match
        matches = self.tools.pattern_match.execute(
            content=current_chunk.content,
            patterns=self.patterns
        )
        
        for match in matches:
            # Step 2: Route
            tier = self._route_extraction(match, current_chunk)
            
            if tier == "skip":
                continue
            
            # Step 3: Extract
            if tier == "pattern":
                extraction = self._fast_extract(match, stack)
            elif tier == "local":
                extraction = self._local_verify(match, stack)
            else:  # cloud
                extraction = self._deep_analyze(stack)
            
            if extraction:
                # Step 4: Suggest relations
                extraction.relation_hints = self.tools.suggest_relations.execute(
                    content=extraction.content,
                    type=extraction.type
                )
                
                # Step 5: Enqueue
                self.tools.enqueue.execute(extraction)
                extractions.append(extraction)
        
        return extractions
    
    def _route_extraction(self, match, chunk) -> str:
        """Decide how to handle this match."""
        if match.confidence > self.thresholds.fast_path:
            return "pattern"
        
        if self._is_high_value(chunk.content):
            return "cloud"
        
        if match.confidence > self.thresholds.local_verify:
            return "local"
        
        if match.confidence > self.thresholds.min_confidence:
            return "pattern"  # Good enough
        
        return "skip"
    
    def _is_high_value(self, content: str) -> bool:
        """Check for high-value trigger phrases."""
        content_lower = content.lower()
        return any(t in content_lower for t in self.high_value_triggers)

            extraction_path="fast",
            speaker=stack.current_chunk.speaker,

class PatternMatchTool(Tool):
    """Fast regex-based pattern detection."""
    
    name = "pattern_match"
    description = "Scan text for extraction trigger patterns"
    
    def execute(self, content: str, patterns: Dict) -> List[Match]:
        matches = []
        content_lower = content.lower()
        
        for node_type, triggers in patterns.items():
            for trigger in triggers:
                if trigger in content_lower:
                    # Calculate confidence based on trigger strength
                    confidence = self._score_match(trigger, content)
                    matches.append(Match(
                        type=node_type,
                        trigger=trigger,
                        confidence=confidence,
                        position=content_lower.find(trigger)
                    ))
        
        return self._dedupe_overlapping(matches)

class TheDude(SmolAgent):
    """
    The Librarian. Files nodes, builds edges, retrieves context.
    
    Tick-driven: processes queue on periodic tick.
    """
    
    name = "the_dude"
    description = "Files extractions, builds graph, retrieves context"
    
    def __init__(self, config: AgentConfig, memory, queue):
        super().__init__(config, memory, queue)
        
        self.tools = [
            FileNodeTool(memory),
            SearchTargetsTool(memory),
            CreateEdgeTool(memory),
            FindOrphansTool(memory),
            ConnectOrphanTool(memory),
            SmartFetchTool(memory),
            HealthCheckTool(memory, queue),
        ]
        
        # Connection strategy (UI-tunable)
        self.strategy = self._load_strategy()
        
        # Tick tracking
        self.ticks_since_maintenance = 0
    
    # ========== TICK HANDLER ==========
    
    def on_tick(self):
        """
        Called by Hub tick loop.
        Adaptive: 250ms active, 5s idle.
        """
        # Process queue
        batch = self.queue.dequeue_batch(self.strategy.batch_size)
        
        for package in batch:
            try:
                self._process_package(package)
                self.queue.ack(package.extraction_id)
            except Exception as e:
                self.queue.nack(package.extraction_id, str(e))
        
        # Check for maintenance
        self.ticks_since_maintenance += 1
        if self.ticks_since_maintenance >= 300:  # ~5 min at active rate
            self.on_maintenance()
            self.ticks_since_maintenance = 0
    
    def _process_package(self, package: ScribedPackage):
        """File node and build edges."""
        
        # Step 1: File the node
        node_id = self.tools.file_node.execute(
            type=package.type,
            content=package.content,
            confidence=package.confidence,
            metadata={
                "speaker": package.speaker,
                "timestamp": package.timestamp,
                "session_id": package.session_id,
                "source_chunk": package.source_chunk_id,
            },
            tags=package.suggested_tags
        )
        
        # Step 2: Build edges from hints
        for hint in package.relation_hints:
            self._try_create_edge(node_id, hint)
    
    def _try_create_edge(self, source_id: str, hint: dict):
        """Search for target and create edge if confident enough."""
        
        # Search for target
        candidates = self.tools.search_targets.execute(
            query=hint["target_query"],
            limit=5
        )
        
        if not candidates:
            return  # No target found, skip
        
        if len(candidates) == 1:
            # Clear match
            target = candidates[0]
            confidence = hint["confidence"]
        else:
            # Ambiguous - use local model or pick best
            if self.config.model in ["local", "cloud"]:
                target, confidence = self._disambiguate(source_id, candidates, hint)
            else:
                target = candidates[0]  # Best FTS5 match
                confidence = hint["confidence"] * 0.8  # Discount for ambiguity
        
        # Check threshold
        if confidence >= self.strategy.edge_confidence_threshold:
            self.tools.create_edge.execute(
                source=source_id,
                target=target.id,
                edge_type=hint["type"],
                strength=confidence / 100,
                metadata={"reason": hint.get("reason", "")}
            )
        elif self.strategy.create_weak_edges and confidence >= self.strategy.weak_edge_threshold:
            # Create weak "related_to" instead
            self.tools.create_edge.execute(
                source=source_id,
                target=target.id,
                edge_type="related_to",
                strength=confidence / 100,
                metadata={"reason": "weak match", "original_type": hint["type"]}
            )
    
    # ========== MAINTENANCE ==========
    
    def on_maintenance(self):
        """Periodic: orphan hunting, health check."""
        
        # Find orphans
        orphans = self.tools.find_orphans.execute(
            limit=self.strategy.orphan_batch_size
        )
        
        for orphan in orphans:
            self._try_connect_orphan(orphan)
        
        # Health check
        health = self.tools.health_check.execute()
        self._update_health_state(health)
    
    def _try_connect_orphan(self, orphan):
        """Try to find connections for an orphan node."""
        
        # Search for related nodes
        candidates = self.tools.search_targets.execute(
            query=orphan.content,
            limit=10,
            exclude=[orphan.id],
            depth=self.strategy.orphan_search_depth
        )
        
        for candidate in candidates:
            similarity = self._compute_similarity(orphan, candidate)
            
            if similarity >= self.strategy.edge_confidence_threshold:
                self.tools.create_edge.execute(
                    source=orphan.id,
                    target=candidate.id,
                    edge_type="related_to",
                    strength=similarity,
                    metadata={"source": "orphan_hunt"}
                )
                break  # One connection is enough to de-orphan
    
    # ========== RETRIEVAL ==========
    
    def retrieve(self, query: str, budget: str = "balanced") -> 'LunaContextPacket':
        """
        On-demand: Luna needs context.
        """
        return self.tools.smart_fetch.execute(
            query=query,
            budget=budget
        )
    
    # ========== MONITORING ==========
    
    def get_status(self) -> Dict:
        """Extended status for UI."""
        base = super().get_status()
        
        health = self.tools.health_check.execute()
        
        return {
            **base,
            "queue": self.queue.stats(),
            "graph": {
                "nodes": health["node_count"],
                "edges": health["edge_count"],
                "orphans": health["orphan_count"],
                "ratio": health["edge_ratio"],
            },
            "strategy": {
                "mode": self.strategy.mode,
                "edge_threshold": self.strategy.edge_confidence_threshold,
                "orphan_depth": self.strategy.orphan_search_depth,
            }
        }

class SmartFetchTool(Tool):
    """Retrieve context for Luna."""
    
    name = "smart_fetch"
    description = "Search and traverse graph for relevant context"
    
    BUDGETS = {
        "minimal": {"fts_limit": 5, "graph_depth": 0, "tokens": 1800},
        "balanced": {"fts_limit": 10, "graph_depth": 1, "tokens": 3800},
        "rich": {"fts_limit": 20, "graph_depth": 2, "tokens": 7200},
    }
    
    def execute(self, query: str, budget: str = "balanced") -> LunaContextPacket:
        params = self.BUDGETS[budget]
        
        # Pass 1: FTS5 search
        anchors = self.memory.fts_search(query, limit=params["fts_limit"])
        
        # Pass 2: Graph traversal (if budget allows)
        connections = []
        if params["graph_depth"] > 0:
            for anchor in anchors[:3]:  # Top 3 anchors
                connected = self.memory.traverse(
                    anchor.id,
                    depth=params["graph_depth"]
                )
                connections.extend(connected)
        
        # Pass 3: Trim to token budget
        return self._build_packet(query, anchors, connections, params["tokens"])
