## The Spin Mechanism

What triggers a spin? What decides which chambers to load?

```python
class RevolvingContext:
    """The cylinder that spins."""
    
    chambers: dict[str, ContextChamber]
    active_chambers: list[str]
    token_budget: int
    
    def spin(self, trigger: SpinTrigger) -> ContextWindow:
        """Load chambers based on what's needed."""
        
        match trigger:
            case TopicShift(new_topic):
                # Spin in relevant memories
                self.load_chamber("memory", query=new_topic)
                
            case DeepReference(entity):
                # Spin in entity's full context
                self.load_chamber("entity", id=entity)
                
            case ModeChange(new_mode):
                # Spin in mode-specific context
                self.load_chamber("mode", mode=new_mode)
        
        return self.build_active_window()
```

## THE SPIN TRIGGERS

What makes each ring rotate?

```python
class ZoetropeContext:
    outer: Ring   # Identity
    middle: Ring  # Session  
    inner: Ring   # Immediate
    
    def on_turn(self, turn: Turn):
        """Every message spins the inner ring."""
        self.inner.spin()
    
    def on_topic_shift(self, new_topic: str):
        """Topic changes spin the middle ring."""
        self.middle.spin_to(slot="topic", content=new_topic)
    
    def on_mode_change(self, mode: str):
        """Hat changes spin the middle ring."""
        self.middle.spin_to(slot="hat", content=mode)
    
    def on_identity_update(self, update: IdentityDelta):
        """Rare. Personality evolution. Spin outer."""
        self.outer.spin_to(slot=update.target, content=update.new_value)
    
    def get_view(self, caliber: str) -> ContextWindow:
        """What Ben sees through the slit."""
        match caliber:
            case ".22": 
                return self.inner.current_frame()
            case ".38": 
                return self.inner.current_frame() + self.middle.current_frame()
            case ".44":
                return self.inner.current_frame() + self.middle.current_frame() + self.outer.current_frame()
```
