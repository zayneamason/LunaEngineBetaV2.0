## QUEUE DATA STRUCTURE

```python
class HandoffQueue:
    """
    The bridge between Ben (extraction) and Dude (filing).
    
    Persistent: Survives daemon restart.
    Priority-aware: High-value extractions processed first.
    Batched: Dude grabs N items at once.
    """
    
    def __init__(self, db_path: str = "data/runtime/handoff_queue.db"):
        self.db = sqlite3.connect(db_path)
        self._init_schema()
    
    def _init_schema(self):
        self.db.execute("""
            CREATE TABLE IF NOT EXISTS handoff_queue (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                extraction_id TEXT UNIQUE,
                payload JSON,
                priority TEXT DEFAULT 'normal',
                status TEXT DEFAULT 'pending',
                enqueued_at REAL,
                processed_at REAL,
                attempts INTEGER DEFAULT 0,
                error TEXT
            )
        """)
    
    # ========== BEN'S INTERFACE ==========
    
    def enqueue(self, package: ScribedPackage) -> str:
        """
        Ben calls this after extraction.
        Returns queue item ID.
        """
        priority = self._compute_priority(package)
        
        self.db.execute("""
            INSERT INTO handoff_queue (extraction_id, payload, priority, enqueued_at)
            VALUES (?, ?, ?, ?)
        """, (
            package["extraction_id"],
            json.dumps(package),
            priority,
            time.time()
        ))
        
        return package["extraction_id"]
    
    def _compute_priority(self, package: ScribedPackage) -> str:
        """
        Determine processing priority.
        """
        # High priority: decisions, high confidence, high value
        if package["type"] == "DECISION":
            return "high"
        if package["confidence"] >= 90:
            return "high"
        if package["quality_flags"]["high_value"]:
            return "high"
        
        # Low priority: low confidence, not novel
        if package["confidence"] < 60:
            return "low"
        if not package["quality_flags"]["is_novel"]:
            return "low"
        
        return "normal"
    
    # ========== DUDE'S INTERFACE ==========
    
    def dequeue_batch(self, max_items: int = 10) -> List[ScribedPackage]:
        """
        Dude calls this on his periodic tick.
        Returns batch of packages, priority-ordered.
        """
        rows = self.db.execute("""
            SELECT id, payload FROM handoff_queue
            WHERE status = 'pending'
            ORDER BY 
                CASE priority 
                    WHEN 'high' THEN 0 
                    WHEN 'normal' THEN 1 
                    WHEN 'low' THEN 2 
                END,
                enqueued_at ASC
            LIMIT ?
        """, (max_items,)).fetchall()
        
        packages = []
        for row in rows:
            # Mark as processing
            self.db.execute("""
                UPDATE handoff_queue 
                SET status = 'processing', attempts = attempts + 1
                WHERE id = ?
            """, (row[0],))
            
            packages.append(json.loads(row[1]))
        
        return packages
    
    def ack(self, extraction_id: str):
        """
        Dude calls this after successful filing.
        Removes from queue.
        """
        self.db.execute("""
            UPDATE handoff_queue 
            SET status = 'completed', processed_at = ?
            WHERE extraction_id = ?
        """, (time.time(), extraction_id))
    
    def nack(self, extraction_id: str, error: str):
        """
        Dude calls this if filing fails.
        Marks for retry.
        """
        self.db.execute("""
            UPDATE handoff_queue 
            SET status = 'pending', error = ?
            WHERE extraction_id = ?
        """, (error, extraction_id))
    
    def dead_letter(self, extraction_id: str, error: str):
        """
        After N failed attempts, move to dead letter.
        """
        self.db.execute("""
            UPDATE handoff_queue 
            SET status = 'dead', error = ?
            WHERE extraction_id = ?
        """, (error, extraction_id))
    
    # ========== MONITORING ==========
    
    def stats(self) -> dict:
        """
        Queue health metrics.
        """
        return {
            "pending": self._count("pending"),
            "processing": self._count("processing"),
            "completed_today": self._count_today("completed"),
            "dead": self._count("dead"),
            "oldest_pending_age": self._oldest_pending_age()
        }
```

## BACKPRESSURE HANDLING

What if Ben is extracting faster than I can file?

```python
class HandoffQueue:
    
    MAX_PENDING = 100  # Alert threshold
    MAX_QUEUE = 500    # Hard limit
    
    def enqueue(self, package: ScribedPackage) -> str:
        pending = self._count("pending")
        
        # Soft limit: warn but accept
        if pending > self.MAX_PENDING:
            log.warning(f"Queue backpressure: {pending} pending")
        
        # Hard limit: drop low priority
        if pending > self.MAX_QUEUE:
            priority = self._compute_priority(package)
            if priority == "low":
                log.warning(f"Dropping low-priority extraction: {package['extraction_id']}")
                return None  # Dropped
            # High/normal still accepted
        
        return self._insert(package)
```

```
Queue < 100:    Normal operation
Queue 100-500:  Warning logged, still accepting
Queue > 500:    Drop LOW priority, accept HIGH/NORMAL
Queue > 1000:   🚨 Something is very wrong
```

## 2. BATCH SIZE

How many items does The Dude grab per tick?

| Batch Size | Throughput | Latency | Memory Spike | Transaction Size |
|------------|------------|---------|--------------|------------------|
| **1** | Low | Best (FIFO) | Minimal | Tiny |
| **5** | Medium | Good | Low | Small |
| **10** | High | Okay | Medium | Medium |
| **25** | Very High | Delayed | Higher | Large |
| **All** | Maximum | Variable | 🔴 Risk | Unbounded |

### The Tradeoff

```
THROUGHPUT ◄─────────────────────────► FAIRNESS

  batch=50                              batch=1
    │                                      │
    ▼                                      ▼
  "Process everything               "Every item gets
   fast, but item #50                immediate attention"
   waits for #1-49"                  
```

### The Math

```
Assume: 10ms to file a node + build edges

Batch=1:   10ms per tick, 1 item processed
Batch=5:   50ms per tick, 5 items processed  
Batch=10:  100ms per tick, 10 items processed
Batch=25:  250ms per tick, 25 items processed

With 0.25s tick interval:
├── Batch=1:  4 items/second max
├── Batch=5:  20 items/second max (but 50ms blocks)
├── Batch=10: 40 items/second max (but 100ms blocks)
└── Batch=25: 100 items/second max (but 250ms blocks!)
```

### The Problem

If batch processing takes longer than tick interval:

```
Tick 0.00s: Start processing batch of 10 (takes 100ms)
Tick 0.25s: Still processing! Skip this tick
Tick 0.50s: Done. Start next batch.

Result: Effective tick rate = 0.5s, not 0.25s
```

### Recommendation: **Adaptive Batch Size**

```python
def get_batch_size(self, tick_interval: float) -> int:
    """
    Size batch to fit within tick interval.
    """
    PROCESSING_TIME_PER_ITEM = 0.010  # 10ms estimate
    SAFETY_MARGIN = 0.8  # Use 80% of tick time max
    
    available_time = tick_interval * SAFETY_MARGIN
    max_items = int(available_time / PROCESSING_TIME_PER_ITEM)
    
    return max(1, min(max_items, 20))  # Clamp 1-20

# Results:
# tick=0.25s → batch=20 (200ms processing, fits in 250ms)
# tick=1.0s  → batch=20 (capped)
# tick=5.0s  → batch=20 (capped)
```

**Or simpler:** Batch=5 with 0.25s tick. Safe, responsive, predictable.

### Failure Scenarios

```
WHY WOULD FILING FAIL?

   └── Retry probably won't help
   
2. Database locked/busy
   └── Retry WILL help (transient)
   
3. Malformed data from Ben
   └── Retry won't help, need fix
   
4. Disk full
   └── Retry won't help, need space
   
5. Bug in Dude's code
   └── Retry won't help, need fix
```

### Recommendation: **Tiered Response**

```python
class FailureHandler:
    
    def handle_failure(self, package, error, attempts):
        error_type = self.classify_error(error)
        
        if error_type == "transient":
            # Database busy, network blip
            if attempts < 3:
                return "retry"  # Try again next tick
            else:
                return "dead_letter"  # Give up, save for review
        
        elif error_type == "data_error":
            # Bad data from Ben
            log.error(f"Malformed package: {package}")
            return "dead_letter"  # Don't retry, won't help
        
        elif error_type == "target_not_found":
            # Edge target doesn't exist
            # File node anyway, skip edges
            return "partial_success"
        
        else:
            # Unknown error
            return "dead_letter"
```

```
DEAD LETTER QUEUE:
├── Stored separately in SQLite
├── Reviewed weekly (or on alert)
├── Can replay after bug fix
├── Never auto-deleted
└── Luna can report: "I have 3 memories I couldn't file"
```

### Alternative: Weighted Scoring

```python
def compute_priority_score(package) -> float:
    """
    Returns 0.0 - 1.0 priority score.
    """
    score = 0.0
    
    # Type weight (0.0 - 0.3)
    type_weights = {
        "DECISION": 0.30,
        "ACTION": 0.25,
        "PROBLEM": 0.20,
        "OUTCOME": 0.15,
        "FACT": 0.10,
        "ASSUMPTION": 0.05
    }
    score += type_weights.get(package.type, 0.1)
    
    # Confidence weight (0.0 - 0.3)
    score += (package.confidence / 100) * 0.3
    
    # Quality flags (0.0 - 0.4)
    if package.quality_flags.high_value:
        score += 0.15
    if package.quality_flags.is_novel:
        score += 0.10
    if package.quality_flags.is_actionable:
        score += 0.10
    if package.quality_flags.is_specific:
        score += 0.05
    
    return min(1.0, score)

def get_priority_tier(score: float) -> str:
    if score >= 0.7:
        return "high"
    elif score >= 0.4:
        return "normal"
    else:
        return "low"
```

### Comparison

| Approach | Simplicity | Tunable | Explainable |
|----------|------------|---------|-------------|
| **If/else rules** | ⚡ Simple | 🔴 Hard | 🟢 Easy |
| **Weighted score** | Medium | 🟢 Easy | 🟡 Medium |
| **ML classifier** | Complex | 🟢 Auto | 🔴 Black box |
