    The secretary that keeps Memory Matrix organized.
    Runs continuously in background.

        self.memory = memory_matrix
        self.weights = {
            'timeline': {},      # How recent?
            'project': {},       # Which project?
            'importance': {},    # User flagged?
            'technical': {},     # Implementation or architecture?
            'historical': {},    # Origin story or recent?
            'relevance': {},     # Search rank?
            'personal_luna': {}, # Does Luna care?
            'personal_user': {}  # Does Ahab care?
        }
    
    def run_background_loop(self):
        """
        Continuously update weights as things change.
        This runs ALL THE TIME in the background.
        """
        while True:
            # Update timeline weights (things get older)
            self.update_timeline_weights()
            
            # Update project weights (project focus shifts)
            self.update_project_weights()
            
            # Update importance (user adds flags)
            self.update_importance_weights()
            
            # Apply weights to Memory Matrix edges
            self.apply_weights_to_graph()
            
            # Sleep a bit
            time.sleep(60)  # Update every minute
    
    def update_timeline_weights(self):
        """
        Recent memories = higher weight.
        Old memories = lower weight.
        """
        for memory in self.memory.all_nodes():
            age_days = (now() - memory.created_at).days
            
            # Exponential decay
            self.weights['timeline'][memory.id] = math.exp(-age_days / 30)
            # 30 days ago = 37% of original weight
            # 60 days ago = 14% of original weight
    
    def update_project_weights(self):
        """
        Current project memories = higher weight.
        Other project memories = lower weight.
        """
        current_project = self.get_current_project()  # e.g., "Luna Beta"
        
        for memory in self.memory.all_nodes():
            if memory.project == current_project:
                self.weights['project'][memory.id] = 1.0
            else:
                self.weights['project'][memory.id] = 0.3
    
    def apply_weights_to_graph(self):
        """
        Update Memory Matrix edge strengths based on weights.
        This makes pathfinding naturally follow high-weight paths.
        """
        for edge in self.memory.all_edges():
            # Combine all 8 weights
            combined_weight = (
                self.weights['timeline'][edge.target] * 0.15 +
                self.weights['project'][edge.target] * 0.20 +
                self.weights['importance'][edge.target] * 0.15 +
                self.weights['technical'][edge.target] * 0.10 +
                self.weights['historical'][edge.target] * 0.10 +
                self.weights['relevance'][edge.target] * 0.10 +
