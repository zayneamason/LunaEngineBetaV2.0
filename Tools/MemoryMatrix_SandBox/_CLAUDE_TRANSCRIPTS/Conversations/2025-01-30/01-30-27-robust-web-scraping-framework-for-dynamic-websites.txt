class CheckpointManager:
    def __init__(self, interval=300):  # 5 minutes default
        self.last_checkpoint = time.time()
        self.interval = interval
        self.checkpoint_path = "checkpoints/"
        
    def should_checkpoint(self, current_state):
        # Time-based checkpoint
        if time.time() - self.last_checkpoint > self.interval:
            return True
            
        # Batch-based checkpoint
        if current_state.processed_items % 100 == 0:  # Every 100 items
            return True
            
        return False
```

def save_checkpoint(state, filename=None):
    if filename is None:
        filename = f"checkpoint_{int(time.time())}.json"
    
    # Save main state
    with open(f"checkpoints/{filename}", 'w') as f:
        json.dump(state.to_dict(), f)
    
    # Save collected data separately to avoid memory issues
    df = pd.DataFrame(state.collected_data)
    df.to_csv(f"checkpoints/data_{filename}.csv", index=False)
    
def load_checkpoint(filename):
    state = ScrapingState()
    
    # Load main state
    with open(f"checkpoints/{filename}", 'r') as f:
        data = json.load(f)
        state.current_url = data['current_url']
        state.processed_urls = set(data['processed_urls'])
        for url in data['pending_urls']:
            state.pending_urls.put(url)
        state.progress_markers = data['progress_markers']
        
    # Load collected data
    df = pd.read_csv(f"checkpoints/data_{filename}.csv")
    state.collected_data = df.to_dict('records')
    
    return state
```
