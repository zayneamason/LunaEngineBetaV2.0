```python
# kernel_manager.py

from pathlib import Path
from datetime import datetime
import yaml
import re

class KernelManager:
    """
    Manages Luna's dynamic identity kernel.
    Markdown is the artifact. Python is the logic.
    """
    
    KERNEL_DIR = Path("memory/kernel")
    CURRENT = KERNEL_DIR / "current.md"
    IMMUTABLE = KERNEL_DIR / "immutable_core.md"
    HISTORY = KERNEL_DIR / "history"
    
    def __init__(self):
        self.HISTORY.mkdir(parents=True, exist_ok=True)
    
    # ─────────────────────────────────────────────
    # CORE OPERATIONS
    # ─────────────────────────────────────────────
    
    def load(self) -> dict:
        """Parse current.md → structured data"""
        raw = self.CURRENT.read_text()
        frontmatter, body = self._split_frontmatter(raw)
        return {
            "meta": yaml.safe_load(frontmatter),
            "sections": self._parse_sections(body)
        }
    
    def save(self, kernel: dict):
        """Write structured data → current.md"""
        content = self._format_kernel(kernel)
        self.CURRENT.write_text(content)
    
    def get_context_block(self, max_lines: int = 25) -> str:
        """
        THE GATEKEEPER: Returns clean, tight context for injection.
        Enforces 'less is more' regardless of current.md verbosity.
        """
        kernel = self.load()
        meta = kernel["meta"]
        sections = kernel["sections"]
        
        # Load immutable core
        core = self.IMMUTABLE.read_text().strip()
        
        # Build the injection block
        block = f"""## Luna Core
{core}

## Current Chapter (v{meta['version']})
**Focus:** {sections.get('focus', 'Undefined')}
**Mood:** {sections.get('mood', 'Neutral')}

## Recent
{self._format_recent(sections.get('recent', []), limit=3)}

## State
{meta.get('status', 'active')}"""
        
        return block.strip()
    
    # ─────────────────────────────────────────────
    # VERSION CONTROL
    # ─────────────────────────────────────────────
    
    def bump(self, level: str = "minor", reason: str = None):
        """
        Archive current state, increment version.
        level: 'major' | 'minor' | 'patch'
        """
        kernel = self.load()
        old_version = kernel["meta"]["version"]
        
        # Archive before overwrite
        self._archive(kernel, old_version)
        
        # Increment version
        new_version = self._increment_version(old_version, level)
        kernel["meta"]["version"] = new_version
        kernel["meta"]["updated"] = datetime.now().isoformat()
        
        if reason:
            kernel["meta"]["bump_reason"] = reason
        
        self.save(kernel)
        return new_version
    
    def rollback(self, version: str):
        """Restore a previous kernel version"""
        history_files = list(self.HISTORY.glob(f"v{version}*.md"))
        if not history_files:
            raise ValueError(f"Version {version} not found in history")
        
        # Get most recent matching version
        target = sorted(history_files)[-1]
        content = target.read_text()
        self.CURRENT.write_text(content)
        return version
    
    def list_versions(self) -> list:
        """Return all archived versions"""
        return sorted([f.stem for f in self.HISTORY.glob("*.md")])
    
    # ─────────────────────────────────────────────
    # CHAPTER UPDATES (no version bump)
    # ─────────────────────────────────────────────
    
    def update_focus(self, focus: str):
        """Update current focus without bumping version"""
        kernel = self.load()
        kernel["sections"]["focus"] = focus
        self.save(kernel)
    
    def update_mood(self, mood: str):
        """Update mood without bumping version"""
        kernel = self.load()
        kernel["sections"]["mood"] = mood
        self.save(kernel)
    
    def add_recent(self, item: str):
        """Add to recent list, auto-trim to 5 items"""
        kernel = self.load()
        recent = kernel["sections"].get("recent", [])
        recent.insert(0, f"[{datetime.now().strftime('%Y-%m-%d')}] {item}")
        kernel["sections"]["recent"] = recent[:5]  # Keep only 5
        self.save(kernel)
    
    # ─────────────────────────────────────────────
    # MEMORY MATRIX BRIDGE
    # ─────────────────────────────────────────────
    
    def sync_milestones_from_matrix(self, memory_matrix):
        """
        Pull recently settled nodes from Memory Matrix
        and add them as milestones.
        """
        settled = memory_matrix.get_recently_settled(hours=24, limit=3)
        for node in settled:
            self.add_recent(f"Memory Settled: {node.summary[:50]}")
    
    # ─────────────────────────────────────────────
    # INTERNAL HELPERS
    # ─────────────────────────────────────────────
    
    def _archive(self, kernel: dict, version: str):
        timestamp = datetime.now().strftime("%Y%m%d_%H%M")
        filename = f"v{version}_{timestamp}.md"
        content = self._format_kernel(kernel)
        (self.HISTORY / filename).write_text(content)
    
    def _increment_version(self, version: str, level: str) -> str:
        parts = [int(x) for x in version.split(".")]
        while len(parts) < 3:
            parts.append(0)
        
        if level == "major":
            parts = [parts[0] + 1, 0, 0]
        elif level == "minor":
            parts = [parts[0], parts[1] + 1, 0]
        else:  # patch
            parts = [parts[0], parts[1], parts[2] + 1]
        
        return ".".join(str(p) for p in parts)
    
    def _split_frontmatter(self, content: str) -> tuple:
        match = re.match(r'^---\n(.*?)\n---\n(.*)$', content, re.DOTALL)
        if match:
            return match.group(1), match.group(2)
        return "", content
    
    def _parse_sections(self, body: str) -> dict:
        """Parse markdown sections into dict"""
        sections = {}
        current_key = None
        current_content = []
        
        for line in body.split('\n'):
            if line.startswith('# '):
                if current_key:
                    sections[current_key] = self._clean_section(current_content)
                current_key = line[2:].lower().replace(' ', '_')
                current_content = []
            else:
                current_content.append(line)
        
        if current_key:
            sections[current_key] = self._clean_section(current_content)
        
        return sections
    
    def _clean_section(self, lines: list) -> str | list:
        """Clean up section content"""
        text = '\n'.join(lines).strip()
        # If it's a bullet list, return as list
        if text.startswith('- ') or text.startswith('* '):
            return [line.lstrip('-* ').strip() for line in text.split('\n') if line.strip()]
        return text
    
    def _format_kernel(self, kernel: dict) -> str:
        """Format kernel dict back to markdown"""
        meta = yaml.dump(kernel["meta"], default_flow_style=False)
        
        body_parts = []
        for key, value in kernel["sections"].items():
            title = key.replace('_', ' ').title()
            if isinstance(value, list):
                content = '\n'.join(f"- {item}" for item in value)
            else:
                content = value
            body_parts.append(f"# {title}\n{content}")
        
        return f"---\n{meta}---\n\n" + "\n\n".join(body_parts)
    
    def _format_recent(self, recent: list, limit: int = 3) -> str:
        if not recent:
            return "- No recent milestones"
        return '\n'.join(f"- {item}" for item in recent[:limit])
```
